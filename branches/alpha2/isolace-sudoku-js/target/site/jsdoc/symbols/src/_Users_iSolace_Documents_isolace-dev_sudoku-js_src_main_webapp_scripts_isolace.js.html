1 if (typeof ISOLACE == "undefined" || !ISOLACE) {
  2 	/**
  3 	 * The ISOLACE global namespace object.  If ISOLACE is already defined, the
  4 	 * existing ISOLACE object will not be overwritten so that defined
  5 	 * namespaces are preserved.
  6 	 * @class ISOLACE
  7 	 * @static
  8 	 */
  9 	var ISOLACE = {};
 10 }

 12 /**
 13  * This code is from the folks at YUI - thanks!
 14  * 
 15  * Returns the namespace specified and creates it if it doesn't exist.
 16  *
 17  * Be careful when naming packages. Reserved words may work in some browsers
 18  * and not others. For instance, the following will fail in Safari:
 19  * <pre>
 20  * ISOLACE.namespace("really.long.nested.namespace");
 21  * </pre>
 22  * This fails because "long" is a future reserved word in ECMAScript
 23  *
 24  * For implementation code that uses YUI, do not create your components
 25  * in the namespaces created by the library.  defined by YUI -- create
 26  * your own (ISOLACE.util, ISOLACE.widget, ISOLACE.lang, ISOLACE.env)
 27  *
 28  * @method namespace
 29  * @static
 30  * @param  {String*} arguments 1-n namespaces to create
 31  * @return {Object}  A reference to the last namespace object created
 32  */
 33 ISOLACE.namespace = function() {
 34 	var a = arguments, o = null, i, j, d;
 35 	for (i = 0; i < a.length; i = i + 1) {
 36 		d = ("" + a[i]).split(".");
 37 		o = ISOLACE;
 38 
 39 		// ISOLACE is implied, so it is ignored if it is included
 40 		for (j = (d[0] == "ISOLACE") ? 1 : 0; j < d.length; j = j + 1) {
 41 			o[d[j]] = o[d[j]] || {};
 42 			o = o[d[j]];
 43 		}
 44 	}
 45 
 46 	return o;
 47 };
