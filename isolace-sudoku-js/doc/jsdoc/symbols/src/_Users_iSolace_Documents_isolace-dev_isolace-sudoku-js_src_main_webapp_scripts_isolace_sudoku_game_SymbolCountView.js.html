  9 ISOLACE.sudoku.SymbolCountView = function() {
 10 };
 11 
 12 /**
 13  * Update symbol count HTML.
 14  * @private
 15  * @method update
 16  * @private {array} state An array representing the current state of the board.
 17  */
 18 ISOLACE.sudoku.SymbolCountView.prototype.update = function(state) {
 19     if(this.revealed === undefined) {
 20         this.getRevealCounts(state);
 21     }
 22     var symbolCount = $SUDOKU_UTIL.getSymbolCount(state);
 23     var html = '';
 24     for( var i = 0; i <= 9; i++) {
 25         if(i == 0) {
 26             html += "<div style='height: 1px; width: 15px; float: left;'>  </div>";
 27         } else {
 28             html += "<img class='fiftyPercent' style='float: left;' src='../images/45/numbers/" + i + "-marker.png' />";
 29         }
 30     }
 31     html += '<br/>';
 32     for( var i = 0; i < symbolCount.length; i++) {
 33         var symbol = i + 1;
 34         var count = symbolCount[i];
 35         var imagePath = '../images/45/numbers/'  + symbol + '-marker.png';
 36         var style = '';
 37         if(count > 8) {
 38             style = 'fiftyPercent';
 39         }
 40         var img = "<img class='" + style + "' style='float: left;' src='" + imagePath + "' />";
 41         var bar = '';
 42         var numRevealed = this.revealed[i];
 43         for( var j = 0; j < count; j++) {
 44             bar += $Renderer.renderCountSymbol(i);
 45         }
 46         html += img + bar + '<br/>';
 47     }
 48     $('#symbolCountView').html(html);
 49 };
 50 
 51 /**
 52  * Update symbol count HTML.
 53  * @private
 54  * @method update
 55  * @private {array} state An array representing the current state of the board.
 56  */
 57 ISOLACE.sudoku.SymbolCountView.prototype.getRevealCounts = function(state) {
 58     this.revealed = [0,0,0,0,0,0,0,0,0];
 59     for( var i = 0; i < state.length; i++) {
 60         var value = state[i];
 61         if(value > 0) {
 62             this.revealed[value-10]++;
 63         }
 64     }
 65 }
