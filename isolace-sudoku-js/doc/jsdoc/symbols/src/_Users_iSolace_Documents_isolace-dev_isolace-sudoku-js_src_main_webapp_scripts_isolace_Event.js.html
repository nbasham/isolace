1 /**
  2  * Constructs an Event object.
  3  * @class Event provides bottlenecks for firing events and binding handlers to
  4  *        events.<br/>
  5  * <br/>Intent:
  6  * <ul>
  7  * <li>Set handler context</li>
  8  * <li>Pass parameters to handler vs. data in an event</li>
  9  * <li>Hide jQuery implementation</li>
 10  * <li>Provide a bottleneck</li>
 11  * </ul>
 12  * 
 13  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
 14  * @version 0.1
 15  * @constructor
 16  */
 17 ISOLACE.Event = function() {
 18 };
 19 
 20 /**
 21  * Event firing bottleneck.
 22  * @method fire
 23  * @param {string} eventType The Event type
 24  * @param {array} argArray An array of parameters to be sent with the event.
 25  */
 26 ISOLACE.Event.prototype.fire = function(eventType, argArray) {
 27     var e = jQuery.Event(eventType);
 28     var message = "Firing event type: '" + eventType + "'";
 29     if(arguments.length > 1) {
 30         assertTrue($.isArray(argArray), "Array expected.");
 31         e.payload = argArray;
 32         message += ' with args: [' + argArray + ']';
 33     }
 34 
 35     $Log.debug(message);
 36     $(document).trigger(e);
 37 };
 38 
 39 /**
 40  * Event handler bottleneck.
 41  * @method handle
 42  * @param {string} eventType The Event type
 43  * @param {object} context The context to use when calling the handler.
 44  * @param {function} callback The function to call when the event is triggered.
 45  */
 46 ISOLACE.Event.prototype.handle = function(eventType, context, callback) {
 47     var message = "Called handler for event type: '" + eventType + "'";
 48     $(document).bind(eventType, function(e) {
 49         if(e.payload !== undefined) {
 50             callback.apply(context, e.payload);
 51             message += ' with args: [' + e.payload + ']';
 52         } else {
 53             callback.call(context, e);
 54         }
 55         $Log.debug(message);
 56     });
 57 };
 58 
 59 if(typeof $Event == "undefined" || !$Event) {
 60     /**
 61      * A singleton instance of ISOLACE.Event automatically created as a
 62      * convenience vs. creating a new ISOLACE.Event for each instance.
 63      * @class ISOLACE.Event
 64      * @static
 65      */
 66     var $Event = new ISOLACE.Event();
 67 }
