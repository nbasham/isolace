1 /**
  2  * Constructs an GameEvent object.
  3  * @class Game event definitions, firers and handlers.
  4  * 
  5  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  6  * @version 0.1
  7  * @constructor
  8  */
  9 ISOLACE.GameEvent = function() {
 10     
 11     // encapsulate private static event types
 12     this.MARK = function() {
 13         return 'ISOLACE.EVENT_MARK';
 14     };
 15     this.MARK_MODE = function() {
 16         return 'ISOLACE.EVENT_MARK_MODE';
 17     };
 18     this.GUESS = function() {
 19         return 'ISOLACE.EVENT_GUESS';
 20     };
 21     this.STATE_CHANGE = function() {
 22         return 'ISOLACE.EVENT_STATE_CHANGE';
 23     };
 24 };
 25 
 26 /**
 27  * Unbind game event handlers.
 28  * @method unbind
 29  */
 30 ISOLACE.GameEvent.prototype.unbind = function() {
 31     $(document).unbind(this.MARK());
 32     $(document).unbind(this.MARK_MODE());
 33     $(document).unbind(this.GUESS());
 34     $(document).unbind(this.STATE_CHANGE());
 35 };
 36 
 37 /**
 38  * Fire GUESS event.
 39  * @method fireGuess
 40  * @param {int} value The value of the guess.
 41  * @param {int} index The cell index of the guess.
 42  */
 43 ISOLACE.GameEvent.prototype.fireGuess = function(value, index) {
 44     $Event.fire(this.GUESS(), [value, index]);
 45 };
 46 
 47 /**
 48  * Bind a handler to the GUESS event. The handler expects two arguments (value, index).
 49  * @method handleGuess
 50  * @param {object} context The context to set when calling the handler (usually this).
 51  * @param {function} f The function you wish to invoke when the event is fired.
 52  */
 53 ISOLACE.GameEvent.prototype.handleGuess = function(context, callback) {
 54     $Event.handle(this.GUESS(), context, callback);
 55 };
 56 
 57 /**
 58  * Fire MARK event.
 59  * @method fireMark
 60  * @param {int} value The value of the mark.
 61  * @param {int} index The cell index of the mark.
 62  */
 63 ISOLACE.GameEvent.prototype.fireMark = function(value, index) {
 64     $Event.fire(this.MARK(), [value, index]);
 65 };
 66 
 67 /**
 68  * Bind a handler to the MARK event. The handler expects two arguments (value, index).
 69  * @method handleMark
 70  * @param {object} context The context to set when calling the handler (usually this).
 71  * @param {function} f The function you wish to invoke when the event is fired.
 72  */
 73 ISOLACE.GameEvent.prototype.handleMark = function(context, callback) {
 74     $Event.handle(this.MARK(), context, callback);
 75 };
 76 
 77 /**
 78  * Fire fireMarkMode event.
 79  * @method fireMarkMode
 80  */
 81 ISOLACE.GameEvent.prototype.fireToggleMarkMode = function() {
 82     $Event.fire(this.MARK_MODE());
 83 };
 84 
 85 /**
 86  * Bind a handler to the MARK_MODE event.
 87  * @method handleToggleMarkMode
 88  * @param {object} context The context to set when calling the handler (usually this).
 89  * @param {function} f The function you wish to invoke when the event is fired.
 90  */
 91 ISOLACE.GameEvent.prototype.handleToggleMarkMode = function(context, callback) {
 92     $Event.handle(this.MARK_MODE(), context, callback);
 93 };
 94 
 95 /**
 96  * Fire STATE_CHANGE event.
 97  * @method fireStateChange
 98  * @param {BoardState} boardState The current state of the board.
 99  */
100 ISOLACE.GameEvent.prototype.fireStateChange = function(boardState) {
101     $Event.fire(this.STATE_CHANGE(), [boardState]);
102 };
103 
104 /**
105  * Bind a handler to the STATE_CHANGE event. The handler expects one argument (BorderState).
106  * @method handleStateChange
107  * @param {object} context The context to set when calling the handler (usually this).
108  * @param {function} f The function you wish to invoke when the event is fired.
109  */
110 ISOLACE.GameEvent.prototype.handleStateChange = function(context, callback) {
111     $Event.handle(this.STATE_CHANGE(), context, callback);
112 };

115     /**
116      * A singleton instance of ISOLACE.GameEvent automatically created as a
117      * convenience vs. creating a new ISOLACE.GameEvent for each instance.
118      * @class ISOLACE.GameEvent

120      */
121     var $GameEvent = new ISOLACE.GameEvent();
