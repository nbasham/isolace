1 /**
  2  * Throw an exception if value is true.
  3  * 
  4  * @method assertTrue
  5  * @param {boolean} value Value to check.
  6  * @param {string} message optional Message to report if assert is thrown.
  7  */
  8 function assertTrue(value, message) {
  9     if(value !== true) {
 10         if(message === undefined) {
 11             message = 'Assertion failed.';
 12         }
 13         $Log.error(message);
 14         throw(message);
 15     }
 16 }

 18 /**
 19  * Throw an exception if value is true.
 20  * 
 21  * @method assertFalse
 22  * @param {boolean} value Value to check.
 23  * @param {string} message optional Message to report if assert is thrown.
 24  */
 25 function assertFalse(value, message) {

 28             message = 'Assertion failed.';
 29         }
 30         $Log.error(message);
 31         throw(message);
 32     }
 33 }

 35 /**
 36  * Throw an exception if value is out of range.
 37  * 
 38  * @method assertInRange
 39  * @param {int} value Value to check.
 40  * @param {int} lower Lower limit of range, inclusive.
 41  * @param {int} upper Upper limit of range, inclusive.
 42  */
 43 function assertInRange(value, lower, upper) {
 44     if(value < lower || value > upper) {
 45         var message = "'" + value + "' out of range (must be between " + lower + " and " + upper + " inclusive).";
 46         $Log.error(message);
 47         throw(message);
 48     }
 49 }

 51 /**
 52  * Throw an exception if value is undefined.
 53  * 
 54  * @method assertDefined
 55  * @param {object} value Value to check.
 56  */
 57 function assertDefined(value) {
 58     if(value === undefined) {
 59         var message = "Undefined value.";
 60         $Log.error(message);
 61         throw(message);
 62     }
 63 }
