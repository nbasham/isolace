1 /**
  2  * Render cells using images.
  3  * @class ImageRenderer
  4  * @namespace ISOLACE.sudoku.view.render
  5  * @constructor
  6  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  7  * @version 0.1
  8 */
  9 ISOLACE.sudoku.ImageRenderer = function() {
 10     this.jcells = [];
 11     for(var i = 0; i < 81; i++) {
 12         var cell = $('#c' + i);
 13         this.jcells.push(cell);
 14     }
 15     $Events.handleStateChange(this, this.render);
 16 };
 17 
 18 /**
 19  * Render all cells based on boardState.
 20  * @method render
 21  * @param {BoardState} boardState The current state of the Sudoku board.
 22  */
 23 ISOLACE.sudoku.ImageRenderer.prototype.render = function(boardState) {
 24     for(var i = 0; i < 81; i++) {
 25         var hasMarker = boardState.hasMarker(i);
 26         if(hasMarker) {
 27             this.renderMarkerCell(boardState, i);
 28         } else {
 29             this.renderCell(boardState, i);
 30         }
 31     }
 32     $Log.debug('Rendered: ' + boardState.state);
 33 };
 34 
 35 /**
 36  * @private
 37  */
 38 ISOLACE.sudoku.ImageRenderer.prototype.renderMarkerCell = function(boardState, index) {
 39     var parentCell = this.jcells[index];
 40     this.setImage(parentCell, '');
 41     var s = '';
 42     for(var markerIndex = 0; markerIndex < 9; markerIndex++) {
 43         var value = markerIndex + 1;
 44         s += "<div class='marker" + index + '-' + value + "'> </div>";
 45     }
 46     parentCell.html(s);
 47 
 48     for(var markerIndex = 0; markerIndex < 9; markerIndex++) {
 49         var value = markerIndex + 1;
 50         var cell = $('.marker' + index + '-' + value);
 51         if(boardState.hasMarkerValue(value, index)) {
 52             if(boardState.conflicts(value, index)) {
 53                 this.setImage(cell, value + '-marker-conflict');
 54             } else {
 55                 this.setImage(cell, value + '-marker');
 56             }
 57         }
 58     }
 59 };
 60 
 61 /**
 62  * @private
 63  */
 64 ISOLACE.sudoku.ImageRenderer.prototype.renderCell = function(boardState, index) {
 65     this.jcells[index].html(' ');
 66     var cell = this.jcells[index];
 67     var value = boardState.getValue(index);
 68     var isEditable = boardState.isEditable(index);
 69     if(isEditable) {
 70         if(boardState.conflicts(value, index)) {
 71             this.setImage(cell, value + '-guess-conflict');
 72         } else {
 73             this.setImage(cell, value + '-guess');
 74         }
 75     } else {
 76         this.setImage(cell, value + '-revealed');
 77     }
 78     if(value === 0) {        
 79         this.setImage(cell, '');
 80     }
 81 };
 82 
 83 /**
 84  * @private
 85  */
 86 ISOLACE.sudoku.ImageRenderer.prototype.setImage = function(cell, imageName) {
 87     var imagePath = '../images/45/numbers/'  + imageName + '.png';
 88     cell.css('background-image', 'url(' + imagePath + ')');
 89     cell.css('background-repeat', 'no-repeat');
 90     cell.css('background-position', 'center center');
 91 };
