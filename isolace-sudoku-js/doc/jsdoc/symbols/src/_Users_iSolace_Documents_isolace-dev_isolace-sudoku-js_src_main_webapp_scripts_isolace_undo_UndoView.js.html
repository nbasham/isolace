1 /**
  2  * Constructs an UndoView object.
  3  * @class UndoView provides a GUI with undo and redo buttons, each fire undo and redo events respectively.
  4  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  5  * @version 0.1
  6  * @constructor
  7  */
  8 ISOLACE.UndoView = function() {
  9     $('#undoView').show();
 10     $('#undoButton').bind('click', function() {
 11         $UndoEvent.fireUndoRequestEvent(false);
 12     });
 13     $('#redoButton').bind('click', function() {
 14         $UndoEvent.fireUndoRequestEvent(true);
 15     });
 16     $UndoEvent.handleUpdateUndoUIEvent(this, this.updateUndoUIEvent);
 17 };
 18 
 19 /**
 20  * Update UI concerns e.g. button state.
 21  * @method updateUndoUIEvent
 22  * @private
 23  * @param {boolean} canUndo True if an undo is possible.
 24  * @param {boolean} canRedo True if a redo is possible.
 25  */
 26 ISOLACE.UndoView.prototype.updateUndoUIEvent = function(canUndo, canRedo) {
 27     $('#undoButton').removeClass('ui-state-active');
 28     if(canUndo) {
 29         $('#undoButton').removeClass('ui-state-disabled').removeAttr('disabled');
 30     } else {
 31         $('#undoButton').addClass('ui-state-disabled').attr('disabled', 'true');
 32     }
 33 
 34     $('#redoButton').removeClass('ui-state-active');
 35     if(canRedo) {
 36         $('#redoButton').removeClass('ui-state-disabled').removeAttr('disabled');
 37     } else {
 38         $('#redoButton').addClass('ui-state-disabled').attr('disabled', 'true');
 39     }
 40 };
