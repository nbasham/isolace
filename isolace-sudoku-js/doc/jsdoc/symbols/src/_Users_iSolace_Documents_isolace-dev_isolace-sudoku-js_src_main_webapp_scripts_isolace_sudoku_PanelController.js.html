1 /**
  2  * Constructs an PanelController object.
  3  * @class PanelController handles the loading, showing and hiding of panels (e.g. game, score, help, etc.).
  4  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  5  * @version 0.1
  6  * @constructor
  7  * @param {options} options The Sudoku options.
  8  */
  9 ISOLACE.PanelController = function(options) {
 10     this.options = options;
 11     this.currentPanel = 0;
 12     
 13     this.panelControllers = [];
 14     var panel;
 15 //    var panel = new ISOLACE.AboutPanel();
 16     this.panelControllers.push(null);
 17 
 18     panel = new ISOLACE.GamePanel(options);
 19     this.panelControllers.push(panel);
 20     
 21 //    panel = new ISOLACE.OptionPanel();
 22     this.panelControllers.push(null);
 23     
 24     panel = new ISOLACE.ScorePanel(options);
 25     this.panelControllers.push(panel);
 26     
 27     this.load(this.options);
 28     var me = this;
 29     $('#tabs').bind('tabsselect', function(event, ui) {
 30         me.changePanel(ui.index);
 31         // Objects available in the function context:
 32         // ui.tab     // anchor element of the selected (clicked) tab
 33         // ui.panel   // element, that contains the selected/clicked tab contents
 34         // ui.index   // zero-based index of the selected (clicked) tab
 35 
 36     });
 37 };
 38 
 39 ISOLACE.PanelController.prototype.ABOUT_PANEL = 0;
 40 ISOLACE.PanelController.prototype.GAME_PANEL = 1;
 41 ISOLACE.PanelController.prototype.OPTIONS_PANEL = 2;
 42 ISOLACE.PanelController.prototype.SCORE_PANEL = 3;
 43 
 44 /**
 45  * Load each panel, if it defined a load function.
 46  * @method load
 47  * @private
 48  * @param {options} options The Sudoku options.
 49  */
 50 ISOLACE.PanelController.prototype.load = function(options) {
 51     $Log.debug('Loading panels');
 52     for( var panelIndex = 0; panelIndex < this.panelControllers.length; panelIndex++) {
 53         var panelContext = this.panelControllers[panelIndex];
 54         if(panelContext !== null && panelContext.load) {
 55             panelContext.load.call(panelContext, options);
 56         }
 57     }
 58 };
 59 
 60 /**
 61  * Get the amount to access for each incorrect guess.
 62  * @method changePanel
 63  * @private
 64  * @param {number} panelIndexToShow The index of the panel about to show.
 65  */
 66 ISOLACE.PanelController.prototype.changePanel = function(panelIndexToShow) {
 67     $Log.debug('Changing current panel to ' + panelIndexToShow);
 68     for( var panelIndex = 0; panelIndex < this.panelControllers.length; panelIndex++) {
 69         var panelContext = this.panelControllers[panelIndex];
 70         if(panelContext !== null) {
 71             if(panelIndexToShow == panelIndex) {
 72                 if(panelContext.show) {
 73                     panelContext.show.call(panelContext, this.options);
 74                 }
 75             } else {
 76                 if(panelContext.hide) {
 77                     panelContext.hide.call(panelContext, this.options);
 78                 }
 79             }
 80         }
 81     }
 82 };
