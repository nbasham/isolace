 89 /**
 90  * Show the panel.
 91  * @method show
 92  * @param {object} options The game options.
 93  */
 94 ISOLACE.ScorePanel.prototype.show = function(options) {
 95     var numRows = this.table.fnGetData().length;
 96     var scores = $Persistence.getScores();
 97     var scoreCount = scores.length;
 98     if(numRows < scoreCount) {
 99         for( var numRows = 0; i < scoreCount; i++) {
100             var score = scores[i];
101             if(score.getScore() < this.personalBest) {
102                 this.personalBest = score.getScore();
103             }
104             var scoreArray = this.scoreToRowData(score);
105             this.table.fnAddData(scoreArray);
106         }
107     }
108     if(scoreCount > 0) {
109         var bestScore = $SUDOKU_UTIL.formatTime(this.personalBest);
110         $("#personalBest").html('<b>Your best score is ' + bestScore + '</b>');
111     }
112 };
113 
114 /**
115  * Convert a Score object to an array of row data.
116  * @method scoreToRowData
117  * @private
118  * @param {Score} score The Score object to convert to data.
119  */
120 ISOLACE.ScorePanel.prototype.scoreToRowData = function(score) {
121     var puzzleId = score.getPuzzleId();
122     var level = score.getPuzzleLevel();
123     var date = new Date(parseInt(score.getDate()));
124     var time = $SUDOKU_UTIL.formatTime(score.getTime());
125     var nuMissed = score.getNumMissed();
126     var penalty = $SUDOKU_UTIL.formatTime(score.getNumMissed() * score.getPenalty());
127     var finalTime = $SUDOKU_UTIL.formatTime(score.getScore());
128     var scoreArray = [];
129     var levelStr = 'Novice';
130     if(level == 1) {
131         levelStr = 'Easy';
132     } else if(level == 2) {
133         levelStr = 'Medium';
134     } else if(level == 3) {
135         levelStr = 'Hard';
136     }
137     scoreArray.push(levelStr + '-' + puzzleId);
138     scoreArray.push(date);
139     scoreArray.push(time);
140     scoreArray.push(nuMissed);
141     scoreArray.push(penalty);
142     scoreArray.push(finalTime);
143 
144     return scoreArray;
145 };
