  3 /**
  4  * Event definitions, firers and handlers. The intent of this class is to:
  5  * <ul>
  6  * <li>Hide jQuery implementation</li>
  7  * <li>Encapsulate arbitrary JavaScript event types</li>
  8  * <li>Provide a bottleneck</li>
  9  * </ul>
 10  * @namespace ISOLACE.sudoku
 11  * @constructor
 12  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
 13  * @version 0.1
 14  * @class Events
 15  */
 16 ISOLACE.sudoku.Events = function() {
 17     // encapsulate private static event types
 18     this.MARK = function() {
 19         return 'ISOLACE.sudoku.EVENT_MARK';
 20     };
 21     this.GUESS = function() {
 22         return 'ISOLACE.sudoku.EVENT_GUESS';
 23     };
 24     this.SHOW_BOARD = function() {
 25         return 'ISOLACE.sudoku.EVENT_BOARD_SHOW';
 26     };
 27     this.SHOW_MAIN_MENU = function() {
 28         return 'ISOLACE.sudoku.EVENT_MAIN_MENU_SHOW';
 29     };
 30     this.SOLVED = function() {
 31         return 'ISOLACE.sudoku.EVENT_SOLVED';
 32     };
 33     this.STATE_CHANGE = function() {
 34         return 'ISOLACE.sudoku.EVENT_STATE_CHANGE';
 35     };
 36 };

 38 /**
 39  * Event handler bottleneck.
 40  * @method handle
 41  * @private
 42  * @param {string} eventType The Event type
 43  * @param {object} eventType The Event type
 44  */
 45 ISOLACE.sudoku.Events.prototype.handle = function(eventType, context, callback) {
 46     $(document).bind(eventType, function(e) {
 47         if(e.payload !== undefined) {
 48         }
 49         callback.call(context, e);
 50     });
 51 };
 52 
 53 /**
 54  * Event firing bottleneck.
 55  * @method fire
 56  * @private
 57  * @param {string} eventType The Event type
 58  * @param {object} eventType The Event type
 59  */
 60 ISOLACE.sudoku.Events.prototype.fire = function(type, payload) {
 61     var e = jQuery.Event(type);
 62     if(payload !== undefined) {
 63         $.extend(e, payload);
 64         // e.payload = payload;
 65     }
 66 
 67     $(document).trigger(e);
 68 };

 70 /**
 71  * Fire GUESS event.
 72  * @method fireGuess
 73  * @param {int} value The value of the guess.
 74  * @param {int} index The cell index of the guess.
 75  */
 76 ISOLACE.sudoku.Events.prototype.fireGuess = function(value, index) {
 77     $Events.fire($Events.GUESS(), {
 78         value : value,
 79         index : index
 80     });
 81 };

 83 /**

 90     $(document).bind($Events.GUESS(), function(e) {
