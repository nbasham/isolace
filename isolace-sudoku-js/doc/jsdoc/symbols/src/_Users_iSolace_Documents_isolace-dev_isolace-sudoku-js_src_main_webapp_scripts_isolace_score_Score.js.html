 39 ISOLACE.Score.PENALTY = 30;
 40 
 41 /**
 42  * Calculate the score based on time and penalty time.
 43  * @method getScore
 44  * @return The total time, in seconds, including penalties.
 45  * @type number
 46  */
 47 ISOLACE.Score.prototype.getScore = function() {
 48     var totalTime = this.getPenalty() * this.numMissed + this.time;
 49     return totalTime;
 50 };
 51 
 52 /**
 53  * Get the amount to access for each incorrect guess.
 54  * @method getPenalty
 55  */
 56 ISOLACE.Score.prototype.getPenalty = function() {
 57     return ISOLACE.Score.PENALTY;
 58 };

147 /**
148  * @method toScoreCsv
149  * @return The comma separated values representing the Score object as a string.
150  * @type string
151  */
152 ISOLACE.Score.prototype.toScoreCsv = function() {
153     var scoreCsvStr = this.getPuzzleId() + ',' + this.getPuzzleLevel() + ',' + this.getDate() + ',' + this.getTime()
154             + ',' + this.getNumMissed();
155     return scoreCsvStr;
156 };
157 
158 /**
159  * @method fromScoreCsv
160  * @param {string} scoreCsvStr The comma separated values to be parsed into the Score object.
161  */
162 ISOLACE.Score.prototype.fromScoreCsv = function(scoreCsvStr) {
163     var s = scoreCsvStr.split(',');
164     this.setPuzzleId(parseInt(s[0]));
165     this.setPuzzleLevel(parseInt(s[1]));
166     this.setDate(parseInt(s[2]));
167     this.setTime(parseInt(s[3]));
168     this.setNumMissed(parseInt(s[4]));
169 };
