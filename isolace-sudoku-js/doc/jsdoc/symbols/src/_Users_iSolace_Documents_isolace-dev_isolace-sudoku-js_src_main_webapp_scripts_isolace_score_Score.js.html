1 /**
  2  * Constructs an Score object.
  3  * @class Score is an immutable object that contains the elements that comprise
  4  *        a score. This will have to be application specific, and Score should
  5  *        perform the role of an interface.
  6  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  7  * @version 0.1
  8  * @constructor
  9  * @param {number} puzzleId The unique puzzle identifier.
 10  * @param {number} date The date/time the puzzle was solved in milliseconds since 1970.
 11  * @param {number} time The number of seconds it took to solve the puzzle.
 12  * @param {number} numMissed The number of incorrect guesses.
 13  */
 14 ISOLACE.Score = function(puzzleId, date, time, numMissed) {
 15     assertNumber(puzzleId, 'puzzleId should be numeric.');
 16     assertNumber(date, 'date should be passed as milliseconds since 1970.')
 17     assertNumber(time, 'time should be numeric, a value representing the number of seconds it took to finsih the game.');
 18     assertNumber(numMissed, 'numMissed id should be numeric.');
 19     this.puzzleId = puzzleId;
 20     this.date = date;
 21     this.time = time;
 22     this.numMissed = numMissed;
 23 };
 24 
 25 /**
 26  * Calculate the score based on time and penalty time.
 27  * @method getScore
 28  * @return The total time, in seconds, including penalties.
 29  * @type number
 30  */
 31 ISOLACE.Score.prototype.getScore = function() {
 32     var totalTime = this.getPenalty() * this.numMissed + this.time;
 33     return totalTime;
 34 };
 35 
 36 /**
 37  * Get the amount to access for each incorrect guess.
 38  * @method getPenalty
 39  */
 40 ISOLACE.Score.prototype.getPenalty = function() {
 41     return 30;
 42 };
 43 
 44 /**
 45  * @method getPuzzleId
 46  * @return The puzzle id this score applies to.
 47  * @type number
 48  */
 49 ISOLACE.Score.prototype.getPuzzleId = function() {
 50     return this.puzzleId;
 51 };
 52 
 53 /**
 54  * @method getDate
 55  * @return The date/time the puzzle was solved in milliseconds since 1970.
 56  * @type number
 57  */
 58 ISOLACE.Score.prototype.getDate = function() {
 59     return this.date;
 60 };
 61 
 62 /**
 63  * @method getTime
 64  * @return The number of seconds it took to solve this puzzle.
 65  * @type number
 66  */
 67 ISOLACE.Score.prototype.getTime = function() {
 68     return this.time;
 69 };
 70 
 71 /**
 72  * @method getNumMissed
 73  * @return The number incorrect guess attempted while solving this puzzle.
 74  * @type number
 75  */
 76 ISOLACE.Score.prototype.getNumMissed = function() {
 77     return this.numMissed;
 78 };
