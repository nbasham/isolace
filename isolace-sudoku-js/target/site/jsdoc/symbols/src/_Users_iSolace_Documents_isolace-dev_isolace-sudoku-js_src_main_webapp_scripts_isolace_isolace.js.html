 18 /**
 19  * This code is from the folks at YUI - thanks!
 20  * 
 21  * Returns the namespace specified and creates it if it doesn't exist.
 22  *
 23  * Be careful when naming packages. Reserved words may work in some browsers
 24  * and not others. For instance, the following will fail in Safari:
 25  * <pre>
 26  * ISOLACE.namespace("really.long.nested.namespace");
 27  * </pre>
 28  * This fails because "long" is a future reserved word in ECMAScript
 29  *
 30  * @method namespace
 31  * @static
 32  * @param  {String*} arguments 1-n namespaces to create
 33  * @return {Object}  A reference to the last namespace object created
 34  */
 35 ISOLACE.namespace = function() {
 36 	var a = arguments, o = null, i, j, d;
 37 	for (i = 0; i < a.length; i = i + 1) {
 38 		d = ("" + a[i]).split(".");
 39 		o = ISOLACE;
 40 
 41 		// ISOLACE is implied, so it is ignored if it is included
 42 		for (j = (d[0] == "ISOLACE") ? 1 : 0; j < d.length; j = j + 1) {
 43 			o[d[j]] = o[d[j]] || {};
 44 			o = o[d[j]];
 45 		}
 46 	}
 47 
 48 	return o;
 49 };

 71 /**
 72  * Normalized logging. Only tested with Firefox.
 73  *
 74  * @method log
 75  * @static
 76  * @param {string} message The message to log.
 77  */
 78 ISOLACE.log = function(message) {
 79     if(ISOLACE.DEBUG) {
 80         if($.browser.mozilla) {
 81             console.log(message);
 82         } else if($.browser.safari) {
 83             window.console.log(message);
 84         } else if($.browser.msie) {
 85             serverLog(message);
 86         }
 87     }
 88 };

 90 /**
 91  * Log a JavaScript message on the server side in the Java console. IE
 92  * does not yet have a logging console.
 93  *
 94  * REQUIRES -jslog program argument
 95  *
 96  * @method serverLog
 97  * @private
 98  * @param  {string} message The message to log.
 99  * @param {object}  jsonObject A JSON object to log.
100  */
101 function serverLog(s, p) {
102     var logParams = {};
103     logParams.message = s;
104     if (p) {
105         var flatP;
106         for ( var property in p) {
107             flatP += property + ': ' + p[property];
108         }
109         window.status = flatP;
110         logParams.message += flatP;
111     }
112     var params = jQuery.param(logParams).replace(/\+/g, "%20");
113     var ajaxParams = {
114         type : "GET",
115         url : '/isolace/log',
116         async : false
117     };
118     ajaxParams.data = params + '&' + Math.random();
119     jQuery.ajax(ajaxParams);
120 }

122 ISOLACE.arrayToString = function(a) {
123     var s = '[';
124     if(a !== undefined && a !== null) {
125         for(var i = 0; i < a.length; i++) {
126             s += a[i];
127             if(i != (a.length -1)) {
128                 s += ', ';
129             }
130         }
131     }
132     s += ']';
133     
134     return s;
135 };

137 /**
138  * Add a bind method to all functions. Stole this from prototype,
139  * when passing a function pointer this will allow you set 'this'
140  * for that function.
141  * <pre>
142  *     Consider applying the event handler 'handler':
143  *     var listener = new YAHOO.util.KeyListener(document, keys, handler);
144  *     When 'handler' is called, this will be set to document. By using bind
145  *     var listener = new YAHOO.util.KeyListener(document, keys, handler.bind(myObject);
146  *     When 'handler' is called, this will be set to myObject.
147  * </pre>
148  *
149  * @method bind
150  * @param  {object} context The scope to wrap a function call to.
151  */
152 Function.prototype.bind = function(context) {
153     var fun = this;
154     return function(){
155       return fun.apply(context, arguments);
156     };
157 };

194  * Throw an exception if value is out of range.
195  * 
196  * @method assertInRange
197  * @param {int} value Value to check.
198  * @param {int} lower Lower limit of range, inclusive.
199  * @param {int} upper Upper limit of range, inclusive.
200  */
201 function assertInRange(value, lower, upper) {
202     if(value < lower || value > upper) {
203         var message = "'" + value + "' out of range (must be between " + lower + " and " + upper + " inclusive).";
204         $Log.error(message);
205         throw(message);
206     }
207 }
