 29 /**
 30  * Update UI concerns e.g. button state.
 31  * @method handleChange
 32  * @private
 33  * @param {boolean} canUndo True if an undo is possible.
 34  * @param {boolean} canRedo True if a redo is possible.
 35  */
 36 ISOLACE.sudoku.UndoView.prototype.handleChange = function(canUndo, canRedo) {
 37     if(canUndo) {
 38         $('#undoButton').removeClass('ui-state-disabled').removeAttr('disabled');
 39     } else {
 40         $('#undoButton').addClass('ui-state-disabled').attr('disabled', 'true');
 41     }
 42     
 43     if(canRedo) {
 44         $('#redoButton').removeClass('ui-state-disabled').removeAttr('disabled');
 45     } else {
 46         $('#redoButton').addClass('ui-state-disabled').attr('disabled', 'true');
 47     }
 48 };
