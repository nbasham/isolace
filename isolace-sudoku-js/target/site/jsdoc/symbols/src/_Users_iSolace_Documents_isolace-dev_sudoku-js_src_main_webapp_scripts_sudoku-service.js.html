 32 /**
 33  * Initialize the Sudoku service. This is done when starting a new game.
 34  * 
 35  * @method initialize
 36  * @exception Invalid problem
 37  * @exception Invalid mask
 38  */
 39 ISOLACE.sudoku.service.prototype.initialize = function(problem, mask) {
 40 	if (problem === undefined || problem.length != 81) {
 41 		throw new Exception("Invalid problem");
 42 	}
 43 	if (mask === undefined) {
 44 		throw new Exception("Invalid mask");
 45 	}
 46 	answer = problem;
 47 	revealedIndexes = mask;
 48 	// window.console.log('revealedIndexes: ' + revealedIndexes);
 49 	solution = [];
 50 	for ( var i = 0; i < problem.length; i++) {
 51 		solution.push(0);
 52 		if ($.inArray(i, mask) != -1) {
 53 			solution[i] = problem[i];
 54 		}
 55 	}
 56 };

 62  * @method getSolution

 64  *         the current board state, else return the int value specified by

 86 ISOLACE.sudoku.service.prototype.guess = function(index, value) {
 87

 92 		} else {

 98 			var a = $s.getGridIndexes(grid);

100 			for(var i = 0; i < 9; i++) {

102 					continue;

105 				window.console.log('Cell ' + [i] + ' value ' + value + ' result: ' + valueAsHintInGrid);
106 				if(valueAsHintInGrid) {

111 			}

186  */
187 ISOLACE.sudoku.service.prototype.getGridFromIndex = function(index) {
188 	var row = $s.getRowFromIndex(index);
189 	var r = 0;
190 	if(row > 5) {
191 		r = 2;
192 	} else if (row > 2) {
193 		r = 1;
194 	}
195 	var col = $s.getColFromIndex(index);
196 	var c = 0;
197 	if(col > 5) {
198 		c = 2;
199 	} else if (col > 2) {
200 		c = 1;
201 	}
202 	var grid = r*3+c;
203 	//window.console.log('index: ' + index + ' row: ' + r + ' col: ' + c + ' grid: ' + grid);
204

224 /**
225  * Utility function to determine if the value is unique in it's row.
226  * @private
227  */
228 ISOLACE.sudoku.service.prototype.uniqueInRow = function(solution, index, value) {
229 	var row = $s.getRowFromIndex(index);
230 	var startIndex = row * 9;
231 	for(var i = 0; i < 9; i++) {
232 		if(index == startIndex + i) {
233 			continue;
234 		}
235 		var currValue = solution[startIndex + i];
236 		if(value == currValue) {
237 			return false;
238 		}
239 	}
240 	return true;
241 };

253 		var currValue = solution[i];
254 		if(value == currValue) {
255 			return false;
256 		}
257 	}
258 	return true;
