 48 /**
 49  * Initialize board.
 50  * @method initBoard
 51  * @private
 52  */
 53 ISOLACE.sudoku.GameController.prototype.initBoard = function() {
 54     var boardView = new ISOLACE.sudoku.BoardView(this.puzzle, this.options);
 55     var jsLintLikesThis = new ISOLACE.sudoku.BoardViewEvents(boardView, this.puzzle);
 56     var boardController = new ISOLACE.sudoku.BoardController(this.puzzle);
 57     boardView.show(boardController.state);
 58     boardView.start();
 59     this.boardController = boardController;
 60     this.boardView = boardView;
 61 };
 62 
 63 /**
 64  * Initialize undo.
 65  * @method initUndo
 66  * @private
 67  */
 68 ISOLACE.sudoku.GameController.prototype.initUndo = function() {
 69     var undoController = new ISOLACE.sudoku.UndoController(this.puzzle.getInitialState());
 70     var undoView = new ISOLACE.sudoku.UndoView(undoController);
 71     $Events.handleUndoUI(undoView, undoView.handleChange);
 72     this.undoController = undoController;
 73 };
 74 
 75 ISOLACE.sudoku.GameController.prototype.handleUndo = function(isRedo) {
 76     var stateArray;
 77     if(isRedo) {
 78         stateArray = this.undoController.redo();
 79     } else {
 80         stateArray = this.undoController.undo();
 81     }
 82     assertDefined(stateArray);
 83     if(stateArray !== null) {
 84         
 85         $Log.debug("Reseting board state because of undo event. " + stateArray);
 86         this.boardController.state = new ISOLACE.sudoku.BoardState(stateArray);
 87         this.boardView.render(this.boardController.state);
 88         $Events.fireUndoUI(this.undoController.canUndo(), this.undoController.canRedo());
 89     } else {
 90         $Log.debug("UI should prevent this from being possible by deactivating when undo/redo is not possible.");
 91     }
 92 };

 94 /**
 95  * Handle a state changed event.
 96  * 
 97  * We update the undo que with the boardState array. Arrays are passed by
 98  * ref and do not have a clone function so we use splice or concat with no
 99  * arguments to copy the array.
100  * @method handleStateChanged
101  */
102 ISOLACE.sudoku.GameController.prototype.handleStateChanged = function(boardState) {
103     if(this.boardController.solved()) {
104         $Events.fireTimerStop();
105         $("#solvedView").dialog( {
106             modal : true,
107             title : 'Puzzle Solved',
108             buttons: { "Ok": function() { $(this).dialog("close"); $Events.fireShowMainMenu(); } }
109         }).html('You solved the puzzle.');
110     } else {
111         var stateArray = boardState.state.concat();
112         this.undoController.add(stateArray);
113     }
114 };
