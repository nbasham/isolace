 17 
 18 /**
 19  * @method getLevel

 23 };
 24 
 25 /**
 26  * @method show
 27  */
 28 ISOLACE.sudoku.MainMenuView.prototype.show = function() {

 36 
 37 /**
 38  * @method hideMenu
