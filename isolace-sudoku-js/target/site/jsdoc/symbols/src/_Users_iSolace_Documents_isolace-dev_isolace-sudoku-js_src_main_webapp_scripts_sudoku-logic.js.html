 27 /**
 28  * Determine if the value conflicts with other guess in the row, column or grid.
 29  * 
 30  * @param {int} index of the cell.
 31  * @param {int} value (1 to 9).
 32  * @method conflicts
 33  * @return {boolean} If the value conflicts with other guess in the row, column or grid.
 34  */
 35 ISOLACE.sudoku.logic.prototype.conflicts = function(index, value) {
 36     if(value === 0) {
 37         return false;
 38     }
 39 	var uir = $logic.uniqueInRow(solution, index, value);
 40 	if(!uir) {
 41 		return true;
 42 	}
 43 	var uic = $logic.uniqueInCol(solution, index, value);
 44 	if(!uic) {
 45 		return true;
 46 	}
 47 	var uig = $logic.uniqueInGrid(solution, index, value);
 48 	if(!uig) {
 49 		return true;
 50 	}
 51 	return false;
 52 };

 54 /**
 55  * Initialize the Sudoku logic. This is done when starting a new game.
 56  * 
 57  * @method initialize
 58  * @exception Invalid problem
 59  * @exception Invalid mask
 60  */
 61 ISOLACE.sudoku.logic.prototype.initialize = function(problem, mask) {
 62 	if (problem === undefined || problem.length != 81) {
 63 		throw new Exception("Invalid problem");
 64 	}
 65 	if (mask === undefined) {
 66 		throw new Exception("Invalid mask");
 67 	}
 68 	answer = problem;
 69 	revealedIndexes = mask;
 70 	// window.console.log('revealedIndexes: ' + revealedIndexes);
 71 	solution = [];
 72 	for ( var i = 0; i < problem.length; i++) {
 73 		solution.push(0);
 74 		if ($.inArray(i, mask) != -1) {
 75 			solution[i] = problem[i];
 76 		}
 77 	}
 78 };

 80 /**
 81  * Get the solution array or int of a given index.
 82  * 
 83  * @param {int optional} index specifies the index of the returned value.
 84  * @method getSolution
 85  * @return {array | int} If index is undefined return an 81 int array defining
 86  *         the current board state, else return the int value specified by
 87  *         index.
 88  * @exception Invalid solution
 89  */
 90 ISOLACE.sudoku.logic.prototype.getSolution = function(index) {
 91 	if (solution === undefined || solution.length != 81) {
 92 		throw new Exception("Invalid solution");
 93 	}
 94 	if (index === undefined) {
 95 		return solution;
 96 	} else {
 97 		return solution[index];
 98 	}
 99 };

101 /**
102  * Called when the user attempts a guess on a cell.
103  * 
104  * @param {int} index of the cell.
105  * @param {int} guess value (1 to 9).
106  * @method guess
107  */
108 ISOLACE.sudoku.logic.prototype.guess = function(index, value) {
109 
110 	if($.inArray(index, revealedIndexes) == -1) {
111 		if(solution[index] == value) {
112 			//window.console.log('Index set to 0');
113 			solution[index] = 0;
114 		} else {
115 			solution[index] = value;
116 			//window.console.log('Index set to ' + value);
117 			//	after a successful guess remove hints with this value from grid
118 			var grid = $logic.getGridFromIndex(index);
119 			//window.console.log('Checking grid ' + grid + ' to see if any hints of value ' + value + ' exist.');
120 			var a = $logic.getGridIndexes(grid);
121 			//window.console.log('Grid indexes: ' + a);
122 			for(var i = 0; i < 9; i++) {
123 				if(index == a[i]) {
124 					continue;
125 				}
126 				var valueAsHintInGrid = this.hasHint(a[i], value);
127 				//window.console.log('Cell ' + [i] + ' value ' + value + ' result: ' + valueAsHintInGrid);
128 				if(valueAsHintInGrid) {
129 					this.setHint(a[i], value);
130 					//window.console.log('Removed hint of value ' + value + ' from cell ' + a[i] + '.');
131 
132 				}
133 			}
134 
135 		}
136 	} else {
137 		//window.console.log('Index ' + index + ' is a revealed index so ignoring this guess.');
138 	}
139 };

141 /**
142  * Determine if the cell specified by index already contains hint value. If
143  * no hint value is specified, returns true if the cell contains one or more
144  * hints.
145  * 
146  * @param {int} index of the cell.
147  * @param {int optional} hint value (1 to 9).
148  * @method hasHint
149  * @return {boolean} If true if the cell of index already has hint of value
150  */
151 ISOLACE.sudoku.logic.prototype.hasHint = function(index, value) {
152     //window.console.log('index=' + index + " value=" + value);
153 	if(solution[index] >= 0) {
154 		return false;
155 	}
156 	if(value === undefined) {
157 	    return solution[index] < 0;
158 	}
159 	var mask = 1 <<  (value-1);
160 	var has = (-1 * solution[index]) & mask;
161 	//window.console.log('mask: ' + mask + ' solution: ' + (solution[index]) + ' &: ' + has);
162 	return has !== 0;
163 };

166 /**
167  * Determine if the cell specified by index already contains value.
168  * 
169  * @param {int} index of the cell.
170  * @param {int} hint value (1 to 9).
171  * @method hasHint
172  * @return {boolean} If true if the cell of index already has hint of value
173  */
174 ISOLACE.sudoku.logic.prototype.setHint = function(index, value) {
175 	if($.inArray(index, revealedIndexes) != -1) {
176 		//window.console.log('revealed index: ' + index);
177 		return;
178 	}
179 	if(solution[index] > 0) {
180 		solution[index] = 0;
181 	}
182 	var hintValue = 1 << (value-1);
183 	if(this.hasHint(index, value)) {
184 		solution[index] += hintValue;
185 		//window.console.log('Cleared hint value because it already existed.');
186 	} else {
187 		solution[index] -= hintValue;
188 		//window.console.log('setHint() hint: ' + value + ' solution: ' + solution[index] + ' value: ' + (-1 * value));
189 	}
190 };

254 /**
255  * Utility function to determine what grid a given index resides in.
256  * @private
257  */
258 ISOLACE.sudoku.logic.prototype.getGridFromIndex = function(index) {
259 	var row = $logic.getRowFromIndex(index);
260 	var r = 0;
261 	if(row > 5) {
262 		r = 2;
263 	} else if (row > 2) {
264 		r = 1;
265 	}
266 	var col = $logic.getColFromIndex(index);
267 	var c = 0;
268 	if(col > 5) {
269 		c = 2;
270 	} else if (col > 2) {
271 		c = 1;
272 	}
273 	var grid = r*3+c;
274 	//window.console.log('index: ' + index + ' row: ' + r + ' col: ' + c + ' grid: ' + grid);
275 	
276 	return grid;
277 };

283 ISOLACE.sudoku.logic.prototype.getGridIndexes = function(gridIndex) {
284 	var grid0Index = [0, 1, 2, 9, 10, 11, 18, 19, 20];
285 	var gridStartIndexs = [0, 3, 6, 27, 30, 33, 54, 57, 60];
286 	var indexes = [];
287 	
288 	for(var i = 0; i < 9; i++) {
289 		indexes.push(grid0Index[i] + gridStartIndexs[gridIndex]);
290 	}
291 	
292 	return indexes;
293 };

295 /**
296  * Utility function to determine if the value is unique in it's row.
297  * @private
298  */
299 ISOLACE.sudoku.logic.prototype.uniqueInRow = function(solution, index, value) {
300 	var row = $logic.getRowFromIndex(index);
301 	var startIndex = row * 9;
302 	for(var i = 0; i < 9; i++) {
303 		if(index == startIndex + i) {
304 			continue;
305 		}
306 		var currValue = solution[startIndex + i];
307 		if(value == currValue) {
308 			return false;
309 		}
310 	}
311 	return true;
312 };
313 
314 /**
315  * Utility function to determine if the value is unique in it's col.
316  * @private
317  */
318 ISOLACE.sudoku.logic.prototype.uniqueInCol = function(solution, index, value) {
319 	var col = $logic.getColFromIndex(index);
320 	for(var i = col; i < 81; i += 9) {
321 		if(index == i) {
322 			continue;
323 		}
324 		var currValue = solution[i];
325 		if(value == currValue) {
326 			return false;
327 		}
328 	}
329 	return true;
330 };
