  2 ISOLACE.LOG = false;

  4 /**
  5  * Wraps a 3rd party logger implementation.
  6  * The current implementation is log4javascript.
  7  * Tried YUI and jsLog, but preffered log4javascript.
  8  * @class Log
  9  * @namespace ISOLACE
 10  * @constructor
 11  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
 12  * @version 0.1
 13  */
 14 ISOLACE.Log = function() {
 15     if(ISOLACE.LOG) {
 16         this.log = log4javascript.getLogger();
 17 
 18         // Create a PopUpAppender with default options
 19         var popUpAppender = new log4javascript.PopUpAppender();
 20 
 21         // Change the desired configuration options
 22         popUpAppender.setFocusPopUp(false);
 23         popUpAppender.setNewestMessageAtTop(true);
 24         var popUpLayout = new log4javascript.PatternLayout("%-5p: %m%n");
 25         popUpAppender.setLayout(popUpLayout);
 26 
 27         // Add the appender to the logger
 28         this.log.addAppender(popUpAppender);
 29 
 30         this.log.debug("Logging intitialized.");
 31     }
 32 
 33 };
 34 
 35 /**
 36  * Log message to console and browser console (if it exists).
 37  * @method log
 38  * @param {string} message The message to log .
 39  * @param {string} level  The level to log message at (e.g. warn).
 40  */
 41 ISOLACE.Log.prototype.log = function(message, level) {
 42     if(ISOLACE.LOG) {
 43         if(level === undefined || typeof (level) != 'string') {
 44             this.log.debug(message);
 45         } else {
 46             level = level.toLowerCase();
 47             if(level == 'debug') {
 48                 this.debug(message);
 49             } else if(level == 'info') {
 50                 this.info(message);
 51             } else if(level == 'warn') {
 52                 this.warn(message);
 53             } else if(level == 'error') {
 54                 this.error(message);
 55             } else {
 56                 this.debug(message);
 57             }
 58         }
 59         this.console(message);
 60     }
 61 };
 62

 64  * Log message to console and browser console (if it exists).

116 ISOLACE.Log.prototype.console = function(message) {
117     if(ISOLACE.LOG) {
118         if($.browser.mozilla) {
119             console.log(message);
120         } else if($.browser.safari) {
121             window.console.log(message);
122         }
123     }
