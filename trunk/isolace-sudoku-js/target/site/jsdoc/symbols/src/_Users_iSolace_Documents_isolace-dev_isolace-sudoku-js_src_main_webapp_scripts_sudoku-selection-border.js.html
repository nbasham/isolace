 36 /**
 37  * Add selection decoration to cell.
 38  * 
 39  * @param {int} Index of cell to decorate.
 40  * @method addSelectionDecoration
 41  */
 42 ISOLACE.sudoku.ui.selection.border.prototype.addSelectionDecoration = function(cell, index) {
 43     var t = cell.offset().top;
 44     var l = cell.offset().left;
 45     var sel = $selectioner.getSelectionDiv();
 46     if(sel === undefined) {
 47         //  cell.css('border-width') doesn't return a value, can I dynamically get border width?
 48         var h = parseInt(cell.css('height')) - 4;
 49         var w = parseInt(cell.css('width')) - 4;
 50         var s = 'position: absolute; color: transparent; top: ' + t + '; left: ' + l + '; width: ' + w + '; height: ' + h;
 51         var html = "<div class='selectedCell' style='" + s + "'> </div>";
 52         $('#someDiv').append(html);
 53         sel = $('.selectedCell');
 54         $selectioner.setSelectionDiv(sel);
 55     } else {
 56         sel.css('top', t);
 57         sel.css('left', l);
 58     }
 59     sel.css('display', 'block');
 60 };
