  3 /**
  4  * BoardController bridges the board UI and Model (BoardPanel and BoardState) applying
  5  * game logic.
  6  * 
  7  * @class BoardController
  8  * @namespace ISOLACE.sudoku.controller
  9  * @constructor
 10  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
 11  * @version 0.1
 12  */
 13 ISOLACE.sudoku.BoardController = function(puzzle) {
 14     this.puzzle = puzzle;
 15     var initialState = puzzle.getInitialState();
 16     this.state = new ISOLACE.sudoku.BoardState(initialState);
 17     $Events.handleGuess(this, this.guess);
 18     $Events.handleMark(this, this.mark);
 19     this.timerController = new ISOLACE.sudoku.TimerController();
 20 };
 21 
 22 /**
 23  * Checks to see if the puzzle has been successfully solved.
 24  * 
 25  * @method solved
 26  * @return {boolean} True if all of the guesses are complete and correct.
 27  */
 28 ISOLACE.sudoku.BoardController.prototype.solved = function() {
 29     var solved = this.state.solved(this.puzzle);
 30     return solved;
 31 };
 32 
 33 /**
 34  * Player took a guess at the given index.
 35  * @private
 36  * @method guess
 37  * @param {int} value Value between 1 and 9 inclusive to check for.
 38  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
 39  */
 40 ISOLACE.sudoku.BoardController.prototype.guess = function(value, index) {
 41     assertInRange(value, 1, 9);
 42     assertInRange(index, 0, 80);
 43     var isEditable = this.state.isEditable(index);
 44     assertTrue(isEditable, "Can't set a value on a revealed cell.");
 45     
 46     this.state.setValue(value, index);
 47     ISOLACE.log('Index ' + index + ' set to ' + this.state.state[index]);
 48     this.removeMarkersFromGrid(value, index);
 49     //$undo.add($logic.getState());
 50     $Events.fireStateChange(this.state);
 51 };
 52 
 53 /**
 54  * After a successful guess remove markers with this value from grid.
 55  * @private
 56  * @method removeMarkersFromGrid
 57  * @param {int} value Value between 1 and 9 inclusive to check for.
 58  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
 59  */
 60 ISOLACE.sudoku.BoardController.prototype.removeMarkersFromGrid = function(value, index) {
 61     var grid = $SUDOKU_UTIL.getGridFromIndex(index);
 62     //ISOLACE.log('Checking grid ' + grid + ' to see if any hints of value ' + value + ' exist.');
 63     var a = $SUDOKU_UTIL.getGridIndexes(grid);
 64     //ISOLACE.log('Grid indexes: ' + a);
 65     for(var i = 0; i < 9; i++) {
 66         var gridIndex = a[i];
 67         if(index == gridIndex) {
 68             continue;
 69         }
 70         var hasMarkerValue = this.state.hasMarkerValue(value, gridIndex);
 71        // ISOLACE.log('Cell ' + gridIndex + ' value ' + value + ' result: ' + hasMarkerValue);
 72         if(hasMarkerValue) {
 73             //  this will toggle the marker to off
 74             this.state.setMarkerValue(value, gridIndex);
 75             ISOLACE.log('Removed marker of value ' + value + ' from cell ' + gridIndex + '.');
 76 
 77         }
 78     }
 79 };
 80 
 81 /**
 82  * Attempt to set a marker at the given index.
 83  * @private
 84  * @method mark
 85  * @param {int} value Value between 1 and 9 inclusive to check for.
 86  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
 87  */
 88 ISOLACE.sudoku.BoardController.prototype.mark = function(value, index) {
 89     assertInRange(value, 1, 9);
 90     assertInRange(index, 0, 80);
 91     var isEditable = this.state.isEditable(index);
 92     assertTrue(isEditable, "Can't set a value on a revealed cell.");
 93 
 94     this.state.setMarkerValue(value, index);
 95     ISOLACE.log('Index ' + index + ' set to ' + this.state.state[index]);
 96     $Events.fireStateChange(this.state);
 97     //$undo.add($logic.getState());
 98 };
