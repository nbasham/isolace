 17 /**
 18  * Gets the symbol palette
 19  * 
 20  * @method getPalette
 21  * @return HTML of the symbol palette
 22  */
 23 ISOLACE.sudoku.ui.render.color.prototype.getPalette = function(top, left) {
 24     var html = "<div class='palette' style='top: " + top + "; left: " + left + ";'>";
 25     for ( var i = 1; i <= 9; i++) {
 26         html += "<div class='symbol " + $renderer.styles[i] + "' onclick='javascript:$ui.guess(" + i + ")'>" + i + "</div>";
 27     }
 28     html += "</div>";
 29 
 30     return html;
 31 };
 32 
 33 /**
 34  * Get the hint palette

 36  * @method getHintPalette

 39 ISOLACE.sudoku.ui.render.color.prototype.getHintPalette = function(top, left) {

 49 /**
 50  * Set cell HTML of a revealed cell.
 51  * 
 52  * @param {int} Index of cell to be acted on.
 53  * @method updateRevealedCell
 54  */
 55 ISOLACE.sudoku.ui.render.color.prototype.updateRevealedCell = function(index, value) {
 56     var selectedCell = $('#c' + index);
 57     var html = "<div class='" + $renderer.styles[value] + " revealed'> </div>";
 58     selectedCell.empty().html(html);
 59 };
 60 
 61 /**
 62  * Set cell HTML for a guess.
 63  * 
 64  * @param {int} Index of cell.
 65  * @param {int} Cell value.
 66  * @param {boolean} Does this value conflict.
 67  * @method updateGuessCell
 68  */
 69 ISOLACE.sudoku.ui.render.color.prototype.updateGuessCell = function(index, value, conflicts) {
 70     var selectedCell = $('#c' + index);
 71     var c = 'guess ' + $renderer.styles[value];
 72     var html = "<div class='" + c + "'> </div>";
 73     if(conflicts) {
 74         selectedCell.addClass('conflict');
 75     } else {
 76         selectedCell.removeClass('conflict');
 77     }
 78     selectedCell.empty().append(html);
 79 };
 80 
 81 /**
 82  * Set cell HTML for hints.
 83  * 
 84  * @param {int} Index of cell to be acted on.
 85  * @method updateHintCell
 86  */
 87 ISOLACE.sudoku.ui.render.color.prototype.updateHintCell = function(index) {
 88     var selectedCell = $('#c' + index);
 89     selectedCell.removeClass('conflict');
 90     var c = 'hint ';// + $renderer.styles[value];
 91     var s = '';
 92     for ( var i = 1; i <= 9; i++) {
 93         if ($ui.sudokuLogic.hasHint(index, i)) {
 94             if ($ui.sudokuLogic.conflicts(index, i)) {
 95                 s += "<div class='" + c + $renderer.styles[i] + "'>X </div>";
 96             } else {
 97                 s += "<div class='" + c + $renderer.styles[i] + "'> </div>";
 98             }
 99         } else {
100             s += "<div class='" + c + $renderer.styles[0] + "'> </div>";
101         }
102 
103     }
104     //window.console.log('Updating hint to: ' + s);
105    // alert('setting cell ' + index + ' to ' + s)
106     selectedCell.empty().append(s);
107 };
