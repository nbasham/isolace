1 ISOLACE.namespace("sudoku");
  2 
  3 /**
  4  * GameController handles game flow.
  5  * @class GameController
  6  * @namespace ISOLACE.sudoku.controller
  7  * @constructor
  8  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  9  * @version 0.1
 10  */
 11 ISOLACE.sudoku.GameController = function() {
 12     var options = {};
 13     this.mainMenuView = new ISOLACE.sudoku.MainMenuView();
 14     this.boardManager = new ISOLACE.sudoku.BoardManager(options);
 15     $Events.handleShowMainMenu(this, this.showMainMenu);
 16     $Events.handleShowBoard(this, this.showBoard);
 17 };
 18 
 19 /**
 20  * Show the main menu.
 21  * @method showMenu
 22  */
 23 ISOLACE.sudoku.GameController.prototype.showMainMenu = function() {
 24     this.mainMenuView.show();
 25 };
 26 
 27 /**
 28  * Hide the main menu.
 29  * @method hideMenu
 30  */
 31 ISOLACE.sudoku.GameController.prototype.hideMainMenu = function() {
 32     this.mainMenuView.hide();
 33 };
 34 
 35 /**
 36  * Show the main menu.
 37  * @method showMenu
 38  */
 39 ISOLACE.sudoku.GameController.prototype.showBoard = function(level) {
 40     this.hideMainMenu();
 41     this.options = {};
 42     this.boardManager.show(this.options);
 43 };
 44 
 45 /**
 46  * Hide the main menu.
 47  * @method hideMenu
 48  */
 49 ISOLACE.sudoku.GameController.prototype.hideBoard = function() {
 50     this.boardManager.hide();
 51 };
