 45 /**
 46  * Increment the second count.
 47  * @private
 48  * @method update
 49  */
 50 ISOLACE.sudoku.TimerView.prototype.update = function(seconds) {
 51     $('#timerView').html('');
 52     var time = $SUDOKU_UTIL.formatTime(seconds);
 53     var timeElements = time.split('');
 54     for( var i = 0; i < timeElements.length; i++) {
 55         var el = timeElements[i];
 56         if(el == ':') {
 57             el = 'seperator';
 58         }
 59         var src = '../images/45/numbers/timer-' + el + '.png';
 60         $('#timerView').append("<img src='" + src + "' />");
 61     }
 62 };
