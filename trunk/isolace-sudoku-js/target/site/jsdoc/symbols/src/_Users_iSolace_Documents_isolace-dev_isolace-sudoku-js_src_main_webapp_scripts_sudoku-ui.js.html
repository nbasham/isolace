 47 /**
 48  * Create Sudoku board HTML string.
 49  * 
 50  * @method getBoardHTML
 51  * @ return {string} Board HTML;
 52  */
 53 ISOLACE.sudoku.ui.prototype.getBoardHTML = function(top, left) {
 54     var cellIndex = 0;
 55     var html = '';
 56     html += "<div id='board' class='board' style='top: " + top + "; left: " + left + ";'>";
 57     for(var row = 0; row < 9; row++) {
 58         html += "<div class='cellrow'>";
 59         for(var col = 0; col < 9; col++) {
 60             html += "<div id='c" + cellIndex + "' class='cell'> </div>";
 61             cellIndex++;
 62         }
 63         html += "</div>";
 64     }
 65     html += "</div>";
 66     
 67     return html;
 68 };

 80 /**
 81  * Entry point into Sudoku UI logic.
 82  * 
 83  * <ul>
 84  * <li>Iterate over each cell</li>
 85  * <ul>
 86  * <li>Add style classes</li>
 87  * <li>Attach click and mouseover events</li>
 88  * </ul>
 89  * <li>Attach key events</li>
 90  * <li>Start timer</li>
 91  * </ul>
 92  * @param {ISOLACE.sudoku.logic} Object to handle Sudoku game logic.
 93  * @param {ISOLACE.sudoku.ui.renderer} Object to handle cell rendering.
 94  * @method startGame
 95  */
 96 ISOLACE.sudoku.ui.prototype.startGame = function(logic) {
 97     $ui.sudokuLogic = logic;
 98 	$('.cell').each( function(i, el) {
 99 
100 		if ($ui.sudokuLogic.getSolution(i) === 0) {
101 			$(el).addClass('unrevealedCell');
102             $(el).bind("mouseenter", {
103                 index :i
104             }, $ui.handleSelectEvent);
105             $(el).bind("mouseleave", {
106                 index :i
107             }, $ui.handleUnselectEvent);
108 		} else {
109 		    $renderer.updateRevealedCell(i, $ui.sudokuLogic.getSolution(i));
110 		}
111 
112 		if ($ui.sudokuLogic.getGridFromIndex(i) % 2 === 0) {
113 			$(el).addClass('evenGrid');
114 		} else {
115 			$(el).addClass('oddGrid');
116 		}
117 
118 		var row = $ui.sudokuLogic.getRowFromIndex(i);
119 		if (row == 2 || row == 5) {
120 			$(el).addClass('rowBorder');
121 		}
122 		var col = $ui.sudokuLogic.getColFromIndex(i);
123 		if (col == 2 || col == 5) {
124 			$(el).addClass('colBorder');
125 		}
126 	});
127 	$ui.mapKeysToEvents();
128 	$ui.startTimer();
129 };

247 /**
248  * Attach a function to a key event. Events
249  * are not propagated. Implementation introduces
250  * a YUI dependency.
251  * 
252  * @param {function} Function to call on key event.
253  * @param {object} Key values to map.
254  * @method setupKeyHandler
255  */
256 ISOLACE.sudoku.ui.prototype.setupKeyHandler = function(f, keys) {
257 	var handler = function(type, args, obj) {
258 		YAHOO.util.Event.stopEvent(args[1]);
259 		f(args[0] - 48);
260 	};
261 	var listener = new YAHOO.util.KeyListener(document, keys, handler);
262 	listener.enable();
263 };
264 
265 /**
266  * Map keys to event handlers. In the future it might be nice
267  * to provide the user with a UI to map the keys.
268  * 
269  * @method mapKeysToEvents
270  */
271 ISOLACE.sudoku.ui.prototype.mapKeysToEvents = function() {
272 	$ui.setupKeyHandler($ui.hint, { shift: true, keys: $ui.KEY_NUMBERS});
273 	$ui.setupKeyHandler($ui.guess, { shift: false, keys: $ui.KEY_NUMBERS});
274 	$ui.setupKeyHandler($ui.moveLeft, {keys: $ui.KEY_LEFT_ARROW});
275 	$ui.setupKeyHandler($ui.moveLeft, { shift: true, keys: $ui.KEY_TAB});
276 	$ui.setupKeyHandler($ui.moveRight, {keys: [$ui.KEY_TAB, $ui.KEY_RIGHT_ARROW]});
277 	$ui.setupKeyHandler($ui.moveUp, {keys: $ui.KEY_UP_ARROW});
278 	$ui.setupKeyHandler($ui.moveUp, { shift: true, keys: $ui.KEY_RETURN});
279 	$ui.setupKeyHandler($ui.moveDown, {keys: [$ui.KEY_RETURN, $ui.KEY_DOWN_ARROW]});
280 	$ui.setupKeyHandler($ui.ignore, {keys: $ui.KEY_BACKSPACE});
281 };

294 /**
295  * Update the game timer. Occurs once a second.
296  * 
297  * @method updateTimer
298  * @see ISOLACE.sudoku.ui.startTimer
299  */
300 ISOLACE.sudoku.ui.prototype.updateTimer = function() {
301 	var timeStr = '';
302 	var min = Math.floor($ui.playTime / 60);
303 	var sec = $ui.playTime % 60;
304 	if(min < 10) {
305 		timeStr = '0' + min;
306 	} else {
307 		timeStr = '' + min;
308 	}
309 	timeStr += ':';
310 	if(sec < 10) {
311 		timeStr += '0' + sec;
312 	} else {
313 		timeStr += '' + sec;
314 	}
315 	$ui.playTime++;
316 	$('.timer').text(timeStr);
317 	
318 };

320 /**
321  * Increment value to the amount and mod at 81. Private
322  * utility used by moveXXX functions;
323  * 
324  * @private
325  * @param {int} Initial value.
326  * @param {int} Amount to increment by.
327  * @method inc
328  * @return {int} Incremented value.
329  */
330 ISOLACE.sudoku.ui.prototype.inc = function(value, amount) {
331 	value += amount;
332 	if (value < 81) {
333 		return value;
334 	} else {
335 		return Math.abs(81 - value);
336 	}
337 };
338 
339 /**
340  * Decrement value to the amount and mod at 0. Private
341  * utility used by moveXXX functions;
342  * 
343  * @private
344  * @param {int} Initial value.
345  * @param {int} Amount to decrement by.
346  * @method dec
347  * @return {int} Decremented value.
348  */
349 ISOLACE.sudoku.ui.prototype.dec = function(value, amount) {
350 	value -= amount;
351 	if (value >= 0) {
352 		return value;
353 	} else {
354 		return 81 + value;
355 	}
356 };

369 /**
370  * Update the HTML for all board cells.
371  * 
372  * @method updateBoard
373  */
374 ISOLACE.sudoku.ui.prototype.updateBoard = function() {
375     for(var index = 0; index < 81; index++) {
376         if(!$ui.sudokuLogic.revealed(index)) {
377             if($ui.sudokuLogic.hasHint(index)) {
378                 $renderer.updateHintCell(index);
379             } else {
380                 var value = $ui.sudokuLogic.getSolution(index);
381                 var conflicts = $ui.sudokuLogic.conflicts(index, value);
382                 $renderer.updateGuessCell(index, value, conflicts);
383             }
384         }
385     }
386 };

388 /**
389  * Handle a user guess.
390  * 
391  * @param {int} Value of user's guess (1 to 9).
392  * @method guess
393  */
394 ISOLACE.sudoku.ui.prototype.guess = function(value) {
395 	var selectedIndex = $ui.sudokuLogic.selectedIndex();
396 	if (selectedIndex < 0) {
397 	    $ui.showMessage('Select a cell before guessing.');
398 		return;
399 	}
400 	if ($ui.sudokuLogic.revealed(selectedIndex)) {
401 		return;
402 	}
403 	$ui.sudokuLogic.guess(selectedIndex, value);
404 	$ui.updateBoard();
405 
406 	if ($ui.sudokuLogic.solved()) {
407 		clearInterval($ui.playTimer);
408         $ui.unselectCell(selectedIndex);
409         //$('.guessCell').each(function() {YAHOO.util.Event.removeListener(this.id);alert(this.id, 'mouseover')});
410 		$ui.showMessage('You did it!');
411 	}
412 };

423 	    $ui.showMessage('Select a cell before choosing a hint.');

427 		return;
428 	}
429 	$ui.sudokuLogic.setHint(selectedIndex, hintValue);
430 	$ui.updateBoard();
431 };
