 17 /**
 18  * Format seconds to mmm:ss.
 19  * @private
 20  * @method formatTime
 21  * @param {int} seconds The seconds to format.
 22  * @return Seconds formatted as mmm:ss.
 23  */
 24 ISOLACE.sudoku.Util.prototype.formatTime = function(seconds) {
 25     var timeStr = '';
 26     var min = Math.floor(seconds / 60);
 27     var sec = seconds % 60;
 28     if(min < 10) {
 29         timeStr = '0' + min;
 30     } else {
 31         timeStr = '' + min;
 32     }
 33     timeStr += ':';
 34     if(sec < 10) {
 35         timeStr += '0' + sec;
 36     } else {
 37         timeStr += '' + sec;
 38     }
 39     
 40     return timeStr;
 41 };

 43 /**
 44  * Utility function to determine what row a given index resides in.
 45  * @private
 46  */
 47 ISOLACE.sudoku.Util.prototype.getHintArray = function(value) {
 48     var a = [];
 49     for ( var i = 1; i <= 9; i++) {
 50         var mask = 1 <<  (i-1);
 51         var has = value & mask;
 52         if (has) {
 53             a.push(i);
 54         }
 55     }
 56     return a;
 57 };

 59 /**
 60  * Checks to see if the puzzle has been successfully solved.
 61  * 
 62  * @method solved
 63  * @param {Puzzle} Current Sudoku puzzle.
 64  * @param {array[int]} 81 values (0 to 9).
 65  * @return {boolean} True if all cells have the correct guess.
 66  */
 67 ISOLACE.sudoku.Util.prototype.solved = function(puzzle, state) {
 68     var answers = puzzle.getValues();
 69     for(var i = 0; i < 81; i++) {
 70         if(state[i] != answers[i]) {
 71             //window.console.log('wrong answer at index: ' + i + ' got '  + solution[i] + ' expected ' + answer[i] + '.');
 72             return false;
 73         }
 74     }
 75     return true;
 76 };

 78 /**
 79  * Determine if the value conflicts with other guess in the row, column or grid.
 80  * 
 81  * @method conflicts
 82  * @param {array[int]} 81 values (0 to 9).
 83  * @param {int} value (1 to 9).
 84  * @param {int} index of the cell.
 85  * @return {boolean} If the value conflicts with other guess in the row, column or grid.
 86  */
 87 ISOLACE.sudoku.Util.prototype.conflicts = function(state, value, index) {
 88     if(value === 0) {
 89         return false;
 90     }
 91     var uir = $SUDOKU_UTIL.uniqueInRow(state, value, index);
 92     if(!uir) {
 93         return true;
 94     }
 95     var uic = $SUDOKU_UTIL.uniqueInCol(state, value, index);
 96     if(!uic) {
 97         return true;
 98     }
 99     var uig = $SUDOKU_UTIL.uniqueInGrid(state, value, index);
100     if(!uig) {
101         return true;
102     }
103     return false;
104 };

121     return i;

124 /**
125  * Utility function to determine what grid a given index resides in.
126  * @private
127  */
128 ISOLACE.sudoku.Util.prototype.getGridFromIndex = function(index) {
129     var row = $SUDOKU_UTIL.getRowFromIndex(index);
130     var r = 0;
131     if(row > 5) {
132         r = 2;
133     } else if (row > 2) {
134         r = 1;
135     }
136     var col = $SUDOKU_UTIL.getColFromIndex(index);
137     var c = 0;
138     if(col > 5) {
139         c = 2;
140     } else if (col > 2) {
141         c = 1;
142     }
143     var grid = r*3+c;
144     //ISOLACE.log('index: ' + index + ' row: ' + r + ' col: ' + c + ' grid: ' + grid);
145     
146     return grid;
147 };

149 /**
150  * Utility function to determine the array of grid indexes associated with a given grid.
151  * @private
152  */
153 ISOLACE.sudoku.Util.prototype.getGridIndexes = function(gridIndex) {
154     var grid0Index = [0, 1, 2, 9, 10, 11, 18, 19, 20];
155     var gridStartIndexs = [0, 3, 6, 27, 30, 33, 54, 57, 60];
156     var indexes = [];
157     
158     for(var i = 0; i < 9; i++) {
159         indexes.push(grid0Index[i] + gridStartIndexs[gridIndex]);
160     }
161     
162     return indexes;
163 };

165 /**
166  * Utility function to determine if the value is unique in it's row.
167  * @private
168  * @param {array[int]} 81 values (0 to 9).
169  * @param {int} value (1 to 9).
170  * @param {int} index of the cell.
171  * @return {boolean} If the value conflicts with other guess in the row.
172  */
173 ISOLACE.sudoku.Util.prototype.uniqueInRow = function(state, value, index) {
174     var row = $SUDOKU_UTIL.getRowFromIndex(index);
175     var startIndex = row * 9;
176     for(var i = 0; i < 9; i++) {
177         if(index == startIndex + i) {
178             continue;
179         }
180         var currValue = state[startIndex + i];
181         if(value == currValue) {
182             return false;
183         }
184     }
185     return true;
186 };
187 
188 /**
189  * Utility function to determine if the value is unique in it's col.
190  * @private
191  * @param {array[int]} 81 values (0 to 9).
192  * @param {int} value (1 to 9).
193  * @param {int} index of the cell.
194  * @return {boolean} If the value conflicts with other guess in the column.
195  */
196 ISOLACE.sudoku.Util.prototype.uniqueInCol = function(state, value, index) {
197     var col = $SUDOKU_UTIL.getColFromIndex(index);
198     for(var i = col; i < 81; i += 9) {
199         if(index == i) {
200             continue;
201         }
202         var currValue = state[i];
203         if(value == currValue) {
204             return false;
205         }
206     }
207     return true;
208 };

210 /**
211  * Utility function to determine if the value is unique in it's grid.
212  * @private
213  * @param {array[int]} 81 values (0 to 9).
214  * @param {int} value (1 to 9).
215  * @param {int} index of the cell.
216  * @return {boolean} If the value conflicts with other guess in the grid.
217  */
218 ISOLACE.sudoku.Util.prototype.uniqueInGrid = function(state, value, index) {
219     var grid = $SUDOKU_UTIL.getGridFromIndex(index);
220     var a = $SUDOKU_UTIL.getGridIndexes(grid);
221     //ISOLACE.log('gridIndexes: ' + a);
222     for(var i = 0; i < 9; i++) {
223         if(index == a[i]) {
224             continue;
225         }
226         var currValue = state[a[i]];
227         if(value == currValue) {
228             return false;
229         }
230     }
231     return true;
232 };
