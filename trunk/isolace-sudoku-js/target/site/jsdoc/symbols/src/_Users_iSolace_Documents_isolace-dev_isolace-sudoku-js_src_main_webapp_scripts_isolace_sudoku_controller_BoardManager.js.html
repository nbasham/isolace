 42 /**
 43  * Initialize board.
 44  * @method initBoard
 45  * @private
 46  */
 47 ISOLACE.sudoku.BoardManager.prototype.initBoard = function() {
 48     var boardView = new ISOLACE.sudoku.BoardView(this.options);
 49     var jsLintLikesThis = new ISOLACE.sudoku.BoardViewEvents(boardView, this.puzzle);
 50     var initialState = this.puzzle.getInitialState();
 51     this.state = new ISOLACE.sudoku.BoardState(initialState);
 52     boardView.show(this.state);
 53     boardView.start();
 54     $Events.handleGuess(this, this.handleGuess);
 55     $Events.handleMark(this, this.handleMark);
 56     this.boardView = boardView;
 57 };

 59 /**
 60  * Initialize undo.
 61  * @method initUndo
 62  * @private
 63  */
 64 ISOLACE.sudoku.BoardManager.prototype.initUndo = function() {
 65     var undoController = new ISOLACE.sudoku.UndoController(this.puzzle.getInitialState());
 66     var undoView = new ISOLACE.sudoku.UndoView(undoController);
 67     $Events.handleUndoUI(undoView, undoView.handleChange);
 68     this.undoController = undoController;
 69 };
 70 
 71 ISOLACE.sudoku.BoardManager.prototype.handleUndo = function(isRedo) {
 72     var stateArray;
 73     if(isRedo) {
 74         stateArray = this.undoController.redo();
 75     } else {
 76         stateArray = this.undoController.undo();
 77     }
 78     assertDefined(stateArray);
 79     if(stateArray !== null) {
 80         
 81         $Log.debug("Reseting board state because of undo event. " + stateArray);
 82         this.state = new ISOLACE.sudoku.BoardState(stateArray);
 83         this.boardView.render(this.state);
 84         $Events.fireUndoUI(this.undoController.canUndo(), this.undoController.canRedo());
 85     } else {
 86         $Log.debug("UI should prevent this from being possible by deactivating when undo/redo is not possible.");
 87     }
 88 };

 90 /**
 91  * Set a guess at the given index.
 92  * @private
 93  * @method handleGuess
 94  * @param {int} value Value between 1 and 9 inclusive to check for.
 95  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
 96  */
 97 ISOLACE.sudoku.BoardManager.prototype.handleGuess = function(value, index) {
 98     this.state.setValue(value, index);

114 /**
115  * Handle a state changed event.
116  * 
117  * We update the undo que with the boardState array. 
118  * @method handleStateChanged
119  */
120 ISOLACE.sudoku.BoardManager.prototype.handleStateChanged = function(boardState) {
121     var solved = this.state.solved(this.puzzle);
122     if(solved) {
123         $Persistence.incPuzzleIndex();
124         $Events.fireTimerStop();
125         $("#solvedView").dialog( {
126             modal : true,
127             title : 'Puzzle Solved',
128             buttons: { "Ok": function() { $(this).dialog("close"); location.reload(); } }
129         }).html('You solved the puzzle.');
130     } else {
131         this.undoController.add(this.state.state);
132     }
133 };
