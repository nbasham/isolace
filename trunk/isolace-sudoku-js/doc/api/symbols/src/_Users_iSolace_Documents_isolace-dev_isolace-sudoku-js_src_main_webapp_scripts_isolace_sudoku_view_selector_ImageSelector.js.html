 13 ISOLACE.sudoku.ImageSelector = function() {
 14     this.selector = undefined;
 15 };
 16 
 17 /**
 18  * Represents a cell being selected by adding a image.
 19  * 
 20  * @method select
 21  */
 22 ISOLACE.sudoku.ImageSelector.prototype.select = function(index) {
 23     if(this.selector === undefined) {
 24         this.createSelectorElement();
 25     }
 26     var cell = $('#c' + index);
 27     var t = cell.offset().top;
 28     var l = cell.position().left;
 29     this.selector.css('top', t);
 30     this.selector.css('left', l);
 31     this.selector.css('display', 'block');
 32 };
 33 
 34 /**
 35  * Not required because select moves the image from the previously selected cell.
 36  * 
 37  * @method unselect
 38  */
 39 ISOLACE.sudoku.ImageSelector.prototype.unselect = function(cell, index) {
 40 };
 41 
 42 
 43 /**
 44  * Create selector HTMLELement.
 45  * @private
 46  * @method createSelectorElement
 47  */
 48 ISOLACE.sudoku.ImageSelector.prototype.createSelectorElement = function() {
 49     this.selector = $('<img/>', {
 50         src: '../images/45/numbers/select.png',
 51         id: 'selector'
 52     }).appendTo('.board');
 53 };
