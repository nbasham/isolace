 13 ISOLACE.sudoku.BorderSelector = function() {
 14     this.div = undefined;
 15 };
 16 
 17 /**
 18  * Represents a cell being selected by adding a border.
 19  * 
 20  * @method select
 21  */
 22 ISOLACE.sudoku.BorderSelector.prototype.select = function(index) {
 23     var cell = $('#c' + index);
 24     var t = cell.offset().top;
 25     var l = cell.offset().left;
 26     if(this.div === undefined) {
 27         //  cell.css('border-width') doesn't return a value, can I dynamically get border width?
 28         var h = parseInt(cell.css('height')) - 4;
 29         var w = parseInt(cell.css('width')) - 4;
 30         var s = 'position: absolute; color: transparent; top: ' + t + '; left: ' + l + '; width: ' + w + 'px; height: ' + h + 'px;';
 31         var html = "<div class='selectedCell' style='" + s + "'> </div>";
 32         $('#boardView').append(html);
 33         this.div = $('.selectedCell');
 34     }
 35     this.div.css('top', t);
 36     this.div.css('left', l);
 37     this.div.css('display', 'block');
 38 };
