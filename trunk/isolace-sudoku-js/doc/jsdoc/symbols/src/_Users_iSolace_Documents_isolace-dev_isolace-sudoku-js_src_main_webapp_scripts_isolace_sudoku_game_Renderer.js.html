  3 /**
  4  * Creates a Renderer object.
  5  * @class Renderer renders symbols as colors.
  6  * @constructor
  7  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  8  * @version 0.1
  9  */
 10 ISOLACE.sudoku.Renderer = function() {
 11     this.renderers = {};
 12     this.renderers.image = new ISOLACE.sudoku.ImageRenderer();
 13     this.renderers.color = new ISOLACE.sudoku.ColorRenderer();
 14     $GameEvent.handleStateChange(this, this.render);
 15 };
 16 
 17 /**
 18  * Render all cells based on boardState.
 19  * @method render
 20  * @param {BoardState} boardState The current state of the Sudoku board.
 21  */
 22 ISOLACE.sudoku.Renderer.prototype.render = function(boardState) {
 23     for(var i = 0; i < 81; i++) {
 24         var cell = $('#c' + i);
 25         cell.css('background-image', '');
 26         var hasMarker = boardState.hasMarker(i);
 27         if(hasMarker) {
 28             this.getRenderer().renderMarkerCell(boardState, i);
 29         } else {
 30             this.getRenderer().renderCell(boardState, i);
 31         }
 32     }
 33     this.renderSymbolCount(boardState.state);
 34     $Log.debug('Rendered: ' + boardState.state);
 35 };
 36 
 37 /**
 38  * @private
 39  */
 40 ISOLACE.sudoku.Renderer.prototype.getRenderer = function() {
 41     var symbolType = $Options.getSymbolType();
 42     if(symbolType == 'color') {
 43         return this.renderers.color;
 44     }
 45     return this.renderers.image;
 46 };
 47 
 48 /**
 49  * Update symbol count HTML.
 50  * @private
 51  * @method renderSymbolCount
 52  * @private {array} state An array representing the current state of the board.
 53  */
 54 ISOLACE.sudoku.Renderer.prototype.renderSymbolCount = function(state) {
 55     if(this.revealed === undefined) {
 56         this.getRevealCounts(state);
 57     }
 58     var symbolCount = $SUDOKU_UTIL.getSymbolCount(state);
 59     var html = '';
 60     for( var i = 0; i <= 9; i++) {
 61         if(i == 0) {
 62             html += "<div class='symbolCountHeader'>  </div>";
 63         } else {
 64             html += "<img class='symbolCountHeader' src='../images/45/numbers/" + i + "-marker.png' />";
 65         }
 66     }
 67     html += '<br/>';
 68     for( var i = 0; i < symbolCount.length; i++) {
 69         var symbol = i + 1;
 70         var count = symbolCount[i];
 71         var imagePath = '../images/45/numbers/'  + symbol + '-marker.png';
 72         var style = '';
 73         if(count > 8) {
 74             style = 'fiftyPercent';
 75         }
 76         var img = "<img class='" + style + "' style='float: left;' src='" + imagePath + "' />";
 77         var bar = '';
 78         var numRevealed = this.revealed[i];
 79         for( var j = 0; j < count; j++) {
 80             bar += this.getRenderer().renderCountSymbol(i);
 81         }
 82         html += img + bar + '<br/>';
 83     }
 84     $('#symbolCountView').html(html);
 85 };
 86 
 87 /**
 88  * Update symbol count HTML.
 89  * @private
 90  * @method update
 91  * @private {array} state An array representing the current state of the board.
 92  */
 93 ISOLACE.sudoku.Renderer.prototype.getRevealCounts = function(state) {
 94     this.revealed = [0,0,0,0,0,0,0,0,0];
 95     for( var i = 0; i < state.length; i++) {
 96         var value = state[i];
 97         if(value > 0) {
 98             this.revealed[value-10]++;
 99         }
100     }
101 }
