 65 /**
 66  * Show the panel.
 67  * @method show
 68  * @param {object} options The game options.
 69  */
 70 ISOLACE.ScorePanel.prototype.show = function(options) {
 71     var numRows = this.table.fnGetData().length;
 72     var scores = $Scores.get();
 73     var scoreCount = scores.length;
 74     if(numRows < scoreCount) {
 75         for( var numRows = 0; i < scoreCount; i++) {
 76             var score = scores[i];
 77             var scoreArray = this.scoreToRowData(score);
 78             this.table.fnAddData(scoreArray);
 79         }
 80     }
 81 };
 82 
 83 
 84 /**
 85  * Convert a Score object to an array of row data.
 86  * @method scoreToRowData
 87  * @private
 88  * @param {Score} score The Score object to convert to data.
 89  */
 90 ISOLACE.ScorePanel.prototype.scoreToRowData = function(score) {
 91     var puzzleId = score.getPuzzleId();
 92     var date = new Date(parseInt(score.getDate()));
 93     var time = $SUDOKU_UTIL.formatTime(score.getTime());
 94     var nuMissed = score.getNumMissed();
 95     var penalty = $SUDOKU_UTIL.formatTime(score.getNumMissed() * score.getPenalty());
 96     var finalTime = $SUDOKU_UTIL.formatTime(score.getScore());
 97     var scoreArray = [];
 98     scoreArray.push(puzzleId);
 99     scoreArray.push(date);
100     scoreArray.push(time);
101     scoreArray.push(nuMissed);
102     scoreArray.push(penalty);
103     scoreArray.push(finalTime);
104     
105     return scoreArray;
106 };
