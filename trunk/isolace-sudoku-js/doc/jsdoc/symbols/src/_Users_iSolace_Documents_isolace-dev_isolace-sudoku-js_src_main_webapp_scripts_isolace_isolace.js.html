 18 /**
 19  * This code is from the folks at YUI - thanks!
 20  * 
 21  * Returns the namespace specified and creates it if it doesn't exist.
 22  *
 23  * Be careful when naming packages. Reserved words may work in some browsers
 24  * and not others. For instance, the following will fail in Safari:
 25  * <pre>
 26  * ISOLACE.namespace("really.long.nested.namespace");
 27  * </pre>
 28  * This fails because "long" is a future reserved word in ECMAScript
 29  *
 30  * @method namespace
 31  * @static
 32  * @param  {String*} arguments 1-n namespaces to create
 33  * @return {Object}  A reference to the last namespace object created
 34  */
 35 ISOLACE.namespace = function() {
 36 	var a = arguments, o = null, i, j, d;
 37 	for (i = 0; i < a.length; i = i + 1) {
 38 		d = ("" + a[i]).split(".");
 39 		o = ISOLACE;
 40 
 41 		// ISOLACE is implied, so it is ignored if it is included
 42 		for (j = (d[0] == "ISOLACE") ? 1 : 0; j < d.length; j = j + 1) {
 43 			o[d[j]] = o[d[j]] || {};
 44 			o = o[d[j]];
 45 		}
 46 	}
 47 
 48 	return o;
 49 };

 71 ISOLACE.arrayToString = function(a) {
 72     var s = '[';
 73     if(a !== undefined && a !== null) {
 74         for(var i = 0; i < a.length; i++) {
 75             s += a[i];
 76             if(i != (a.length -1)) {
 77                 s += ', ';
 78             }
 79         }
 80     }
 81     s += ']';
 82     
 83     return s;
 84 };

 86 /**
 87  * Check if two arrays are equal. This is only implemented for arrays containing primitive values.
 88  * 
 89  * @param {array} a1 One of the arrays to compare.
 90  * @param {array} a2 One of the arrays to compare.
 91  * @method arrayEquals
 92  * @return {boolean} True if the array values are equal.
 93  */
 94 function arrayEquals(a1, a2) {
 95     if(a1 === undefined || a2 === undefined || a1.length === undefined || a2.length === undefined) {
 96         return false;
 97     }
 98     if(a1.length != a2.length) {
 99         return false;
100     }
101     for(var i = 0; i < a1.length; i++) {
102         if(a1[i] != a2[i]) {
103             return false;
104         }
105     }
106 
107     return true;
108 }

110 /**
111  * Add a bind method to all functions. Stole this from prototype,
112  * when passing a function pointer this will allow you set 'this'
113  * for that function.
114  * <pre>
115  *     Consider applying the event handler 'handler':
116  *     var listener = new YAHOO.util.KeyListener(document, keys, handler);
117  *     When 'handler' is called, this will be set to document. By using bind
118  *     var listener = new YAHOO.util.KeyListener(document, keys, handler.bind(myObject);
119  *     When 'handler' is called, this will be set to myObject.
120  * </pre>
121  *
122  * @method bind
123  * @param  {object} context The scope to wrap a function call to.
124  */
125 Function.prototype.bind = function(context) {
126     var fun = this;
127     return function(){
128       return fun.apply(context, arguments);
129     };
130 };
