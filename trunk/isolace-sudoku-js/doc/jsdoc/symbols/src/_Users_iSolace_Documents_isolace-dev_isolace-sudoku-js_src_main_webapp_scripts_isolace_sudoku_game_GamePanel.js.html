 13 /**
 14  * Initialize the panel.
 15  * @method load
 16  * @param {object} options The game options.
 17  */
 18 ISOLACE.GamePanel.prototype.load = function(options) {
 19     $Events.handleStateChange(this, this.handleStateChanged);
 20     this.puzzle = this.getNextPuzzle();
 21     this.symbolCountView = new ISOLACE.sudoku.SymbolCountView();
 22     this.timerController = new ISOLACE.TimerController();
 23     var jsLintLikesThis1 = new ISOLACE.TimerView();
 24     this.initBoard();
 25     this.initUndo();
 26 };

 46 /**
 47  * Initialize board.
 48  * @method initBoard
 49  * @private
 50  */
 51 ISOLACE.GamePanel.prototype.initBoard = function(options) {
 52     this.numMissed = 0;
 53     var boardView = new ISOLACE.sudoku.BoardView(options);
 54     var jsLintLikesThis = new ISOLACE.sudoku.BoardViewEvents(boardView, this.puzzle);
 55     var initialState = this.puzzle.getInitialState();
 56     this.state = new ISOLACE.sudoku.BoardState(this.puzzle.getValues(), initialState);
 57     boardView.show(this.state);
 58     this.symbolCountView.update(initialState);
 59     boardView.start();
 60     $TimerEvent.fireTimerStart();
 61     $Events.handleGuess(this, this.handleGuess);
 62     $Events.handleMark(this, this.handleMark);
 63     this.boardView = boardView;
 64 };
 65 
 66 /**
 67  * Initialize undo.
 68  * @method initUndo
 69  * @private
 70  */
 71 ISOLACE.GamePanel.prototype.initUndo = function() {
 72     var undoController = new ISOLACE.UndoController(this.puzzle.getInitialState());
 73     var undoView = new ISOLACE.UndoView();
 74     $UndoEvent.handleUndoEvent(this, function(stateArray) {
 75         this.state = new ISOLACE.sudoku.BoardState(this.puzzle.getValues(), stateArray);
 76         this.boardView.render(this.state);
 77         this.symbolCountView.update(stateArray);
 78     });
 79 };
 80 
 81 /**
 82  * Set a guess at the given index.
 83  * @private
 84  * @method handleGuess
 85  * @param {int} value Value between 1 and 9 inclusive to check for.
 86  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
 87  */
 88 ISOLACE.GamePanel.prototype.handleGuess = function(value, index) {
 89     var correctGuess = this.state.setValue(value, index);
 90     if(!correctGuess) {
 91         this.numMissed++;
 92     }
 93     $Events.fireStateChange(this.state);
 94 };
 95 
 96 /**
 97  * Set a marker at the given index.
 98  * @private
 99  * @method mark
100  * @param {int} value Value between 1 and 9 inclusive to check for.
101  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
102  */
103 ISOLACE.GamePanel.prototype.handleMark = function(value, index) {
104     this.state.setMarkerValue(value, index);
105     $Events.fireStateChange(this.state);
106 };
107 
108 /**
109  * Handle a state changed event.
110  * 
111  * We update the undo queue with the boardState array. 
112  * @method handleStateChanged
113  */
114 ISOLACE.GamePanel.prototype.handleStateChanged = function(boardState) {
115     this.symbolCountView.update(this.state.state);
116     var solved = this.state.solved();
117     if(solved) {
118         var index = $Persistence.getPuzzleIndex();
119         var time = this.timerController.getSeconds();
120         var score = $Scores.add(index, time, this.numMissed);
121         var formattedScore = $SUDOKU_UTIL.formatTime(score.getScore());
122         
123         $Persistence.incPuzzleIndex();
124         $TimerEvent.fireTimerStop();
125         $("#solvedView").dialog( {
126             modal : true,
127             title : 'Puzzle Solved',
128             buttons: { "Ok": function() { $(this).dialog("close"); location.reload(); } }
129         }).html('You solved the puzzle. Your score is ' + formattedScore);
130     } else {
131         $UndoEvent.fireSubmitUndoRecordEvent(this.state.state);
132     }
133 };
