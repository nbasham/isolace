 14 /**
 15  * Get the HTML to represent a symbol in the symbol count view.
 16  * @method renderCountSymbol
 17  * @param {int} symbolIndex The index of the symbol to render.
 18  */
 19 ISOLACE.sudoku.ColorRenderer.prototype.renderCountSymbol = function(symbolIndex) {
 20     var countSymbol = "<div class='colorSymbolCount " + this.styles[symbolIndex+1] + "'> </div>";
 21     return countSymbol;

 24 /**
 25  */
 26 ISOLACE.sudoku.ColorRenderer.prototype.renderMarkerCell = function(boardState, index) {
 27     var parentCell = $('#c' + index);
 28     parentCell.css('background-color', 'transparent');
 29     parentCell.empty();
 30     var s = '';
 31     for( var markerIndex = 0; markerIndex < 9; markerIndex++) {
 32         var value = markerIndex + 1;
 33         s += "<div style='margin: 1px;' class='marker marker" + index + '-' + value + "'> </div>";
 34     }
 35     parentCell.html(s);
 36 
 37     for( var markerIndex = 0; markerIndex < 9; markerIndex++) {
 38         var value = markerIndex + 1;
 39         var cell = $('.marker' + index + '-' + value);
 40         if(boardState.hasMarkerValue(value, index)) {
 41             cell.addClass(this.styles[value]);
 42             if(boardState.conflicts(value, index)) {
 43                 cell.html("<div class='colorMarkerConflict'> </div>");
 44             }
 45         }
 46     }
 47 };

 53     for(var index = 0; index < 81; index++) {
 54         var cell = $('#c' + index);
 55         cell.css('background-image', '');
 56         cell.html(' ');
 57         var value = pauseState[index];
 58         if(value === 0) {
 59             cell.css('background-color', 'transparent');
 60         } else {
 61             cell.css('background-color', '#3B5998');
 62         }
 63 
 64         var s = "<div> </div>";
 65         cell.empty().append(s);
 66     }

 69 /**
 70  */
 71 ISOLACE.sudoku.ColorRenderer.prototype.renderCell = function(boardState, index) {
 72     var cell = $('#c' + index);
 73     cell.html(' ');
 74     var value = boardState.getValue(index);
 75     var isEditable = boardState.isEditable(index);
 76     var klass = 'color ' + this.styles[value] + ' ';
 77     cell.css('background-color', 'transparent');
 78     if(isEditable) {
 79         if(boardState.conflicts(value, index)) {
 80             cell.css('background-color', 'red');
 81         }
 82     } else {
 83         klass += ' colorRevealed';
 84         if($.browser.msie) {
 85             klass += 'IE';
 86         }
 87     }
 88 
 89     var s = "<div class='" + klass + "'> </div>";
 90     cell.empty().append(s);
 91 };
