 59 /**
 60  * Event handler bottleneck.
 61  * @method handle
 62  * @private
 63  * @param {string} eventType The Event type
 64  * @param {object} eventType The Event type
 65  */
 66 ISOLACE.sudoku.Events.prototype.handle = function(eventType, context, callback) {
 67     $(document).bind(eventType, function(e) {
 68         var args = [];
 69         if(e.payload !== undefined) {
 70             // make an array from e.payload to pass as arg array to apply
 71             // PROBLEM: e.payload is obj and don't know how to order arg array
 72         }
 73         callback.apply(context, args);
 74     });
 75 };

 77 /**
 78  * Event firing bottleneck.
 79  * @method fire
 80  * @private
 81  * @param {string} eventType The Event type
 82  * @param {object} eventType The Event type
 83  */
 84 ISOLACE.sudoku.Events.prototype.fire = function(type, payload) {
 85     var e = jQuery.Event(type);
 86     if(payload !== undefined) {
 87         $.extend(e, payload);
 88         // e.payload = payload;
 89     }
 90 
 91     $(document).trigger(e);
 92 };

 94 /**
 95  * Fire GUESS event.
 96  * @method fireGuess
 97  * @param {int} value The value of the guess.
 98  * @param {int} index The cell index of the guess.
 99  */
100 ISOLACE.sudoku.Events.prototype.fireGuess = function(value, index) {
101     $Events.fire($Events.GUESS(), {
102         value : value,
103         index : index
104     });
105 };
106 
107 /**
108  * Bind a handler to the GUESS event. The handler expects two arguments (value, index).
109  * @method handleGuess
110  * @param {object} context The context to set when calling the handler (usually this).
111  * @param {function} f The function you wish to invoke when the event is fired.
112  */
113 ISOLACE.sudoku.Events.prototype.handleGuess = function(context, f) {
114     $(document).bind($Events.GUESS(), function(e) {
115         f.call(context, e.value, e.index);
116     });
117 };
