 18 };

 20 /**
 21  * Show the panel.

 23  */
 24 ISOLACE.OptionPanel.prototype.show = function() {
 25     this.setPuzzleLevelButtonState();
 26 };

 33 };

 35 
 36 /**
 37  * @method handleLevelChange
 38  * @private
 39  */
 40 ISOLACE.OptionPanel.prototype.handleLevelChange = function(level) {
 41     this.setPuzzleLevelButtonState();
 42 };
 43 
 44 /**
 45  * Set initial puzzle level button state.
 46  * @method setPuzzleLevelButtonState
 47  * @private
 48  */
 49 ISOLACE.OptionPanel.prototype.setPuzzleLevelButtonState = function() {
 50     $('#level0Button').removeClass('ui-state-active');
 51     $('#level1Button').removeClass('ui-state-active');
 52     $('#level2Button').removeClass('ui-state-active');
 53     $('#level3Button').removeClass('ui-state-active');
 54     var level = $Persistence.getPuzzleLevel();
 55     $('#level' + level + 'Button').addClass('ui-state-active');
 56 };
