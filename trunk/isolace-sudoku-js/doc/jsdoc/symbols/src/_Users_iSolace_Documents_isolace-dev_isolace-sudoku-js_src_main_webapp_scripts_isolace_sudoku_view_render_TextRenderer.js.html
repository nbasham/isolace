  3 /**
  4  * A renderer that updates renders text for each Sudoku board cell.
  5  * 
  6  * @namespace ISOLACE.sudoku.view.render
  7  * @constructor
  8  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  9  * @version 0.1
 10  * @class TextRenderer
 11  */
 12 ISOLACE.sudoku.TextRenderer = function() {
 13     this.jcells = [];
 14     for(var i = 0; i < 81; i++) {
 15         var cell = $('#c' + i);
 16         this.jcells.push(cell);
 17     }
 18 };
 19 
 20 /**
 21  * Render all cells based on boardState.
 22  * @method render
 23  * @param {BoardState} boardState The current state of the Sudoku board.
 24  */
 25 ISOLACE.sudoku.TextRenderer.prototype.render = function(boardState) {
 26     for(var i = 0; i < 81; i++) {
 27 //        var jCell = this.jcells[i];
 28 //        this.jcells[i].html(boardState.state[i]);
 29         var hasMarker = boardState.hasMarker(i);
 30         if(hasMarker) {
 31             this.renderMarkerCell(boardState, i);
 32         } else {
 33             this.renderCell(boardState, i);
 34         }
 35     }
 36 };
 37 
 38 /**
 39  * @private
 40  */
 41 ISOLACE.sudoku.TextRenderer.prototype.renderMarkerCell = function(boardState, index) {
 42     var s = '';
 43     for(var markerIndex = 0; markerIndex < 9; markerIndex++) {
 44         var value = markerIndex + 1;
 45         if(boardState.hasMarkerValue(value, index)) {
 46             if(boardState.conflicts(value, index)) {
 47                 s += "<div class='marker unrevealedCellConflicted'>" + value + "</div>";
 48             } else {
 49                 s += "<div class='marker unrevealedCell'>" + value + "</div>";
 50             }
 51         } else {
 52             s += "<div class='marker'> </div>";
 53         }
 54     }
 55     this.jcells[index].html(s);
 56 };
 57 
 58 /**
 59  * @private
 60  */
 61 ISOLACE.sudoku.TextRenderer.prototype.renderCell = function(boardState, index) {
 62     var jCell = this.jcells[index];
 63     var value = boardState.getValue(index);
 64     var isEditable = boardState.isEditable(index);
 65     if(isEditable) {
 66         jCell.addClass('unrevealedCell');
 67         if(boardState.conflicts(value, index)) {
 68             jCell.addClass('unrevealedCellConflicted');
 69         } else {
 70             jCell.removeClass('unrevealedCellConflicted');
 71         }
 72     } else {
 73         jCell.addClass('revealedCell');
 74     }
 75     if(value === 0) {
 76         value = '';
 77     }
 78     jCell.html(value);
 79 };
