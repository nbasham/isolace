 13 ISOLACE.sudoku.BoardViewEvents = function(boardView, puzzle) {
 14     this.KEY_RETURN = 13;
 15     this.KEY_TAB = 9;
 16     this.KEY_BACKSPACE = 8;
 17     this.KEY_LEFT_ARROW = 37;
 18     this.KEY_RIGHT_ARROW = 39;
 19     this.KEY_UP_ARROW = 38;
 20     this.KEY_DOWN_ARROW = 40;
 21     /**
 22      * 1 to 9 inclusive including key pad
 23      */
 24     this.KEY_NUMBERS = [49,50,51,52,53,54,55,56,57,97,98,99,100,101,102,103,104,105];
 25     /**
 26      * don't map properly on Mac
 27      */
 28     this.KEY_FUNCTION_NUMBERS = [112,113,114,115,116,117,118,119,120];
 29     this.boardView = boardView;
 30     this.selectedCellIndex = -1;
 31     
 32     var me = this;
 33     $('.cell').each( function(i, el) {
 34         var editable = puzzle.isEditable(i);
 35         if (editable) {
 36             $(el).bind("mouseenter", {
 37                 index :i,
 38                 me: me
 39             }, me.handleSelectEvent);
 40             $(el).bind("mouseleave", {
 41                 index :i,
 42                 me: me
 43             }, me.handleUnselectEvent);
 44         }
 45     });
 46     this.mapKeysToEvents();
 47 };

 49 /**
 50  * Map keys to event handlers. In the future it might be nice
 51  * to provide the user with a UI to map the keys.
 52  * 
 53  * @method mapKeysToEvents
 54  */
 55 ISOLACE.sudoku.BoardViewEvents.prototype.mapKeysToEvents = function() {
 56     this.setupKeyHandler(this.mark.bind(this), { shift: true, keys: this.KEY_NUMBERS});
 57     this.setupKeyHandler(this.guess.bind(this), { shift: false, keys: this.KEY_NUMBERS});
 58     this.setupKeyHandler(this.moveLeft.bind(this), {keys: this.KEY_LEFT_ARROW});
 59     this.setupKeyHandler(this.moveLeft.bind(this), { shift: true, keys: this.KEY_TAB});
 60     this.setupKeyHandler(this.moveRight.bind(this), {keys: [this.KEY_TAB, this.KEY_RIGHT_ARROW]});
 61     this.setupKeyHandler(this.moveUp.bind(this), {keys: this.KEY_UP_ARROW});
 62     this.setupKeyHandler(this.moveUp.bind(this), { shift: true, keys: this.KEY_RETURN});
 63     this.setupKeyHandler(this.moveDown.bind(this), {keys: [this.KEY_RETURN, this.KEY_DOWN_ARROW]});
 64     this.setupKeyHandler(jQuery.noop, {keys: this.KEY_BACKSPACE});
 65 };

 75 /**
 76  * @private
 77  * Attach a function to a key event. Events
 78  * are not propagated. Implementation introduces
 79  * a YUI dependency.
 80  * 
 81  * @param {function} Function to call on key event.
 82  * @param {object} Key values to map.
 83  * @method setupKeyHandler
 84  */
 85 ISOLACE.sudoku.BoardViewEvents.prototype.setupKeyHandler = function(f, keys) {
 86     var handler = function(type, args, obj) {
 87         //YAHOO.util.Event.stopEvent(args[1]);
 88         f(args[0] - 48);
 89     };
 90     var listener = new YAHOO.util.KeyListener(document, keys, handler);
 91     listener.enable();
 92 };

211 /**
212  * Increment value to the amount and mod at 81. Private
213  * utility used by moveXXX functions;
214  * 
215  * @private
216  * @param {int} Initial value.
217  * @param {int} Amount to increment by.
218  * @method inc
219  * @return {int} Incremented value.
220  */
221 ISOLACE.sudoku.BoardViewEvents.prototype.inc = function(value, amount) {
222     value += amount;
223     if (value < 81) {
224         return value;
225     } else {
226         return Math.abs(81 - value);
227     }
228 };
229 
230 /**
231  * Decrement value to the amount and mod at 0. Private
232  * utility used by moveXXX functions;
233  * 
234  * @private
235  * @param {int} Initial value.
236  * @param {int} Amount to decrement by.
237  * @method dec
238  * @return {int} Decremented value.
239  */
240 ISOLACE.sudoku.BoardViewEvents.prototype.dec = function(value, amount) {
241     value -= amount;
242     if (value >= 0) {
243         return value;
244     } else {
245         return 81 + value;
246     }
247 };

249 /**
250  * Gets index from event and calls selectCell.
251  * 
252  * @param {object} YUI event.
253  * @method handleSelectEvent
254  */
255 ISOLACE.sudoku.BoardViewEvents.prototype.handleSelectEvent = function(e) {
256     var me = e.data.me;
257     me.selectedCellIndex = e.data.index;
258     me.boardView.selector.select(me.selectedCellIndex);
259 };
