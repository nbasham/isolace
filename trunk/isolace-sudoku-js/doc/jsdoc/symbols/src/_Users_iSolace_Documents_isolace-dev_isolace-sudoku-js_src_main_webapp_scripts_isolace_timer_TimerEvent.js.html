1 /**
  2  * Constructs an TimerEvent object.
  3  * @class Timer event definitions, firers and handlers.
  4  * 
  5  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  6  * @version 0.1
  7  * @constructor
  8  */
  9 ISOLACE.TimerEvent = function() {
 10     
 11     // encapsulate private static event types
 12     this.TIMER_INCREMENT = function() {
 13         return 'ISOLACE.TIMER_INCREMENT';
 14     };
 15     this.TIMER_PAUSE = function() {
 16         return 'ISOLACE.TIMER_PAUSE';
 17     };
 18     this.TIMER_START = function() {
 19         return 'ISOLACE.TIMER_START';
 20     };
 21     this.TIMER_STOP = function() {
 22         return 'ISOLACE.TIMER_STOP';
 23     };
 24     this.TIMER_UNPAUSE = function() {
 25         return 'ISOLACE.TIMER_UNPAUSE';
 26     };
 27 };
 28 
 29 /**
 30  * Unbind timer event handlers.
 31  * @method unbind
 32  */
 33 ISOLACE.TimerEvent.prototype.unbind = function() {
 34     $(document).unbind(this.TIMER_INCREMENT());
 35     $(document).unbind(this.TIMER_PAUSE());
 36     $(document).unbind(this.TIMER_START());
 37     $(document).unbind(this.TIMER_STOP());
 38     $(document).unbind(this.TIMER_UNPAUSE());
 39 };
 40 
 41 /**
 42  * @method fireTimerIncrement
 43  * @param {int} seconds The number of seconds elapsed.
 44  */
 45 ISOLACE.TimerEvent.prototype.fireTimerIncrement = function(seconds) {
 46     $Event.fire(this.TIMER_INCREMENT(), [seconds]);
 47 };
 48 
 49 /**
 50  * @method handleTimerIncrement
 51  * @param {object} context The context to set when calling the handler (usually this).
 52  * @param {function} callback The function to invoke when the event is fired.
 53  */
 54 ISOLACE.TimerEvent.prototype.handleTimerIncrement = function(context, callback) {
 55     $Event.handle(this.TIMER_INCREMENT(), context, callback);
 56 };
