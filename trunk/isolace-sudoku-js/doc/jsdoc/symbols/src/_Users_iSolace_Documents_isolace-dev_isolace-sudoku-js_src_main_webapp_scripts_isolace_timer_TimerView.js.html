1 /**
  2  * Create a TimerView object.
  3  * @class Handles all HTML UI around the timer panel. Styles e.g. position and
  4  *        size are controlled by CSS class 'timer'. HTMLElement with id 'timer'
  5  *        is updated each second.
  6  * @namespace ISOLACE
  7  * @constructor
  8  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
  9  * @version 0.1
 10  */
 11 ISOLACE.TimerView = function() {
 12     $TimerEvent.handleTimerIncrement(this, this.update);
 13     $('#timerView')
 14             .html(
 15                     "<img src='../images/45/numbers/timer-0.png' /><img src='../images/45/numbers/timer-0.png' /><img src='../images/45/numbers/timer-seperator.png' /><img src='../images/45/numbers/timer-0.png' /><img src='../images/45/numbers/timer-0.png' />");
 16 };
 17 
 18 /**
 19  * Increment the second count.
 20  * @private
 21  * @method update
 22  */
 23 ISOLACE.TimerView.prototype.update = function(seconds) {
 24     $('#timerView').html('');
 25     var time = $SUDOKU_UTIL.formatTime(seconds);
 26     var timeElements = time.split('');
 27     for( var i = 0; i < timeElements.length; i++) {
 28         var el = timeElements[i];
 29         if(el == ':') {
 30             el = 'seperator';
 31         }
 32         var src = '../images/45/numbers/timer-' + el + '.png';
 33         $('#timerView').append("<img src='" + src + "' />");
 34     }
 35 };
