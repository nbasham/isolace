  3 /**
  4  * State defines the cell values of a Sudoku board at some point in time.
  5  * Each cell can contain:
  6  *  -   a revealed cell with a value from 10 to 18
  7  *  -   a guessed cell with a value from 1 to 9, 0 if empty (no guess yet)
  8  *  -   a marker(s) cell with a values less than 0
  9  * 
 10  * <pre>
 11  * Example usage:
 12  *      var puzzle = new ISOLACE.sudoku.Puzzle(values, revealedIndexes);
 13  *      var initialState = puzzle.getInitialState();
 14  *      var state = new ISOLACE.sudoku.BoardState(initialState);
 15  * </pre>
 16  * 
 17  * @class State
 18  * @namespace ISOLACE.sudoku.model
 19  * @constructor
 20  * @param {int[]} state The current puzzle state.
 21  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
 22  * @version 0.1
 23  */
 24 ISOLACE.sudoku.BoardState = function(solution, state) {
 25     this.solution = solution;
 26     this.state = state;
 27 };
 28 
 29 /**
 30  * Normalize state by removing conventions applied to cell values by BoardState.
 31  * 
 32  * @method normalize
 33  * @return {array[int]} An array values 0 to 9 inclusive describing the state of the board.
 34  */
 35 ISOLACE.sudoku.BoardState.prototype.equals = function(bs) {
 36     if(bs === undefined || bs === null || bs.state.length != this.state.length) {
 37         return false;
 38     }
 39     var equal = arrayEquals(bs.state, this.state);
 40 
 41     return equal;
 42 };
 43 
 44 /**
 45  * @private
 46  * Normalize state by removing conventions applied to cell values by BoardState.
 47  * 
 48  * @method normalize
 49  * @return {array[int]} An array values 0 to 9 inclusive describing the state of the board.
 50  */
 51 ISOLACE.sudoku.BoardState.prototype.normalize = function() {
 52     var stateArray = [];
 53     for(var i = 0; i < 81; i++) {
 54         var cell = this.state[i];
 55         if(cell < 0) {
 56             cell = 0;
 57         } if (cell > 9) {
 58             cell -= 9;
 59         }
 60         stateArray.push(cell);
 61     }
 62 
 63     return stateArray;
 64 };
 65 
 66 /**
 67  * Checks to see if the puzzle has been successfully solved.
 68  * 
 69  * @method solved
 70  * @param {Puzzle} puzzle The current Sudoku Puzzle.
 71  * @return {boolean} True if all cells have the correct guess.
 72  */
 73 ISOLACE.sudoku.BoardState.prototype.solved = function() {
 74     var stateArray = this.normalize();
 75     var equal = arrayEquals(stateArray, this.solution);
 76     return equal;
 77 };

 79 /**
 80  * Determine if the value conflicts with other guess in the row, column or grid.
 81  * 
 82  * @method conflicts
 83  * @param {int} value Value between 1 and 9 inclusive to check for.
 84  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
 85  * @return {boolean} If the value conflicts with other guess in the row, column or grid.
 86  */
 87 ISOLACE.sudoku.BoardState.prototype.conflicts = function(value, index) {
 88     if(value === 0 || !this.isEditable(index) ) {
 89         return false;
 90     }
 91     assertInRange(value, 1, 9);
 92     assertInRange(index, 0, 80);
 93     var stateArray = this.normalize();
 94     var conflicts = $SUDOKU_UTIL.conflicts(stateArray, value, index);
 95     var wrongAnswer = false;
 96     var rawValue = this.state[index];
 97     if(rawValue > 0) {
 98         wrongAnswer = this.solution[index] != rawValue;
 99     }
100 
101     return conflicts || wrongAnswer;
102 };

104 /**
105  * After a successful guess remove markers with this value from grid.
106  * @private
107  * @method removeMarkersFromGrid
108  * @param {int} value Value between 1 and 9 inclusive to check for.
109  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
110  */
111 ISOLACE.sudoku.BoardState.prototype.removeMarkersFromGrid = function(value, index) {
112     var grid = $SUDOKU_UTIL.getGridFromIndex(index);
113     var a = $SUDOKU_UTIL.getGridIndexes(grid);
114     for(var i = 0; i < 9; i++) {
115         var gridIndex = a[i];
116         if(index == gridIndex) {
117             continue;
118         }
119         var hasMarkerValue = this.hasMarkerValue(value, gridIndex);
120         if(hasMarkerValue) {
121             //  this will toggle the marker to off
122             this.setMarkerValue(value, gridIndex);
123             $Log.debug('Removed marker of value ' + value + ' from cell ' + gridIndex + '.');
124         }
125     }
126 };

128 /**
129  * Set cell to value, if that value already exists remove it.
130  * 
131  * @method setValue
132  * @param {int} value Value between 1 and 9 inclusive to check for.
133  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
134  * @return True if guess is correct, false if incorrect.
135  */
136 ISOLACE.sudoku.BoardState.prototype.setValue = function(value, index) {
137     assertInRange(value, 1, 9);
138     assertInRange(index, 0, 80);
139     var isEditable = this.isEditable(index);
140     assertTrue(isEditable, "Can't set a value on a revealed cell.");
141     if(this.state[index] == value) {
142         $Log.debug('Clear index ' + index + ' of value ' + value);
143         this.state[index] = 0;
144     } else {
145         this.state[index] = value;
146         $Log.debug('Set index ' + index + ' to value ' + value);
147     }
148     this.removeMarkersFromGrid(value, index);
149     var correctGuess = this.solution[index] == value || this.solution[index] === 0;
150     return correctGuess;
151 };

153 /**
154  * Get the value at the given index.
155  * 
156  * @method getValue
157  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
158  * @return {int} Value of the cell at the given index, between 1 and 9 inclusive.
159  */
160 ISOLACE.sudoku.BoardState.prototype.getValue = function(index) {
161     assertInRange(index, 0, 80);
162     if(this.isEditable(index)){
163         return this.state[index];
164     } else {
165         return this.state[index] - 9;
166     }
167 };

169 /**
170  * Determine if the cell at the given index is editable (i.e. wasn't revealed
171  * to the user).
172  * 
173  * @method isEditable
174  * @return {boolean} True if the cell was not originally revealed.
175  */
176 ISOLACE.sudoku.BoardState.prototype.isEditable = function(index) {
177     assertInRange(index, 0, 80);
178     return this.state[index] < 10;
179 };
180 
181 /**
182  * Determine if the cell at the given index contains one or more markers.
183  * 
184  * @method hasMarker
185  * @return {boolean} True if the cell contains one or more markers.
186  */
187 ISOLACE.sudoku.BoardState.prototype.hasMarker = function(index) {
188     assertInRange(index, 0, 80);
189     return this.state[index] < 0;
190 };
191 
192 /**
193  * Update cell value to reflect a new marker value, if that value
194  * already exists remove it.
195  * 
196  * @method setMarkerValue
197  * @private
198  * @param {int} value Value between 1 and 9 inclusive to check for.
199  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
200  */
201 ISOLACE.sudoku.BoardState.prototype.setMarkerValue = function(value, index) {
202     assertInRange(value, 1, 9);
203     assertInRange(index, 0, 80);
204     var isEditable = this.isEditable(index);
205     assertTrue(isEditable, "Can't set a marker on a revealed cell.");
206     
207     //  clear existing guess
208     if(this.state[index] > 0) {
209         this.state[index] = 0;
210     }
211     var mask = 1 <<  (value-1);
212     if(this.hasMarkerValue(value, index)) {
213         this.state[index] += mask;
214         $Log.debug('Clear marker at index ' + index + ' with value ' + value);
215     } else {
216         this.state[index] -= mask;
217         $Log.debug('Set marker at index ' + index + ' to value ' + value);
218     }
219 };

221 /**
222  * Determine if the cell at the given index has a marker of the given value.
223  * 
224  * @method hasMarkerValue
225  * @param {int} value Value between 1 and 9 inclusive to check for.
226  * @param {int} index Index (0 to 80 inclusive) of the cell to check.
227  * @return {boolean} True if the cell at the given index has a marker of the given value.
228  */
229 ISOLACE.sudoku.BoardState.prototype.hasMarkerValue = function(value, index) {
230     assertInRange(value, 1, 9);
231     assertInRange(index, 0, 80);
232     if(this.hasMarker(index)) {
233         var mask = 1 <<  (value-1);
234         var has = (-1 * this.state[index]) & mask;
235         return has;
236     }
237     return false;
238 };
