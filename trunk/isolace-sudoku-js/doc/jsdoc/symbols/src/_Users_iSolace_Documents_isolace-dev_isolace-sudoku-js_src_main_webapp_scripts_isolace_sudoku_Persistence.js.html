  3 /**
  4  * Wraps persistence implementation. For now using cookies, will migrate to 
  5  * a server side solution using Google's AppEngine.
  6  * @class Persistence
  7  * @namespace ISOLACE.sudoku
  8  * @constructor
  9  * @author Norman Basham, iSolace, Copyright (c) 2009-2010. ALL RIGHTS RESERVED<br/>
 10  * @version 0.1
 11 */
 12 ISOLACE.sudoku.Persistence = function() {
 13     var FIFTY_YEARS = 365*50;
 14     this.options = {
 15         expires: FIFTY_YEARS
 16     };
 17 };
 18 
 19 /**
 20  * Internal bottleneck to get a value.
 21  * @method get
 22  * @private
 23  * @param {string} key 
 24  * @return {*} The value, null if not found.
 25  */
 26 ISOLACE.sudoku.Persistence.prototype.get = function(key) {
 27     var value = $.cookie(key);
 28     if(value === '') {
 29         value = null;
 30     }
 31     $Log.debug('Read [' + key + '] = ' + value);
 32     return value;
 33 };
 34 
 35 /**
 36  * Internal bottleneck to set a value.
 37  * @method set
 38  * @private
 39  * @param {string} key 
 40  * @param {*} value 
 41  */
 42 ISOLACE.sudoku.Persistence.prototype.set = function(key, value) {
 43     $.cookie(key, '' + value, this.options);
 44     $Log.debug('Wrote [' + key + '] = ' + value);
 45 };
 46 
 47 /**
 48  * Internal bottleneck to delete a cookie.
 49  * @method remove
 50  * @private
 51  * @param {string} key 
 52  */
 53 ISOLACE.sudoku.Persistence.prototype.remove = function(key) {
 54     $.cookie(key, '', this.options);
 55     $Log.debug('Deleted cookie [' + key + '] = ');
 56 };
 57 
 58 /**
 59  * @method getPuzzleIndex
 60  * @return {int} The current puzzle index.
 61  */
 62 ISOLACE.sudoku.Persistence.prototype.getPuzzleIndex = function() {
 63     var puzzleIndex = this.get('puzzleIndex');
 64     if(puzzleIndex === null) {
 65         puzzleIndex = 0;
 66         this.set('puzzleIndex', puzzleIndex);
 67     }
 68     return parseInt(puzzleIndex);
 69 };
 70 
 71 /**
 72  * 
 73  * @method incPuzzleIndex
 74  * @param {int} puzzleIndex 
 75  */
 76 ISOLACE.sudoku.Persistence.prototype.incPuzzleIndex = function() {
 77     var puzzleIndex = this.getPuzzleIndex() + 1;
 78     this.set('puzzleIndex', puzzleIndex);
 79 };

 84      * convenience vs. creating a new ISOLACE.sudoku.Persistence for each instance.
 85      * @class ISOLACE.sudoku.Persistence
 86      * @static
 87      */
 88     var $Persistence = new ISOLACE.sudoku.Persistence();
